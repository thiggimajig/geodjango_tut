184 def get_height_for_line( 

185 self, 

186 lineno: int, 

187 width: int, 

188 get_line_prefix: Optional[GetLinePrefixCallable], 

189 slice_stop: Optional[int] = None, 

190 ) -> int: 

191 """ 

192 Return the height that a given line would need if it is rendered in a 

193 space with the given width (using line wrapping). 

194 

195 :param get_line_prefix: None or a `Window.get_line_prefix` callable 

196 that returns the prefix to be inserted before this line. 

197 :param slice_stop: Wrap only "line[:slice_stop]" and return that 

198 partial result. This is needed for scrolling the window correctly 

199 when line wrapping. 

200 :returns: The computed height. 

201 """ 

202 # Instead of using `get_line_prefix` as key, we use render_counter 

203 # instead. This is more reliable, because this function could still be 

204 # the same, while the content would change over time. 

205 key = get_app().render_counter, lineno, width, slice_stop 

206 

207 try: 

208 return self._line_heights_cache[key] 

209 except KeyError: 

210 if width == 0: 

211 height = 10 ** 8 

212 else: 

213 # Calculate line width first. 

214 line = fragment_list_to_text(self.get_line(lineno))[:slice_stop] 

215 text_width = get_cwidth(line) 

216 

217 if get_line_prefix: 

218 # Add prefix width. 

219 text_width += fragment_list_width( 

220 to_formatted_text(get_line_prefix(lineno, 0)) 

221 ) 

222 

223 # Slower path: compute path when there's a line prefix. 

224 height = 1 

225 

226 # Keep wrapping as long as the line doesn't fit. 

227 # Keep adding new prefixes for every wrapped line. 

228 while text_width > width: 

229 height += 1 

230 text_width -= width 

231 

232 fragments2 = to_formatted_text( 

233 get_line_prefix(lineno, height - 1) 

234 ) 

235 prefix_width = get_cwidth(fragment_list_to_text(fragments2)) 

236 

237 if prefix_width >= width: # Prefix doesn't fit. 

238 height = 10 ** 8 

239 break 

240 

241 text_width += prefix_width 

242 else: 

243 # Fast path: compute height when there's no line prefix. 

244 try: 

245 quotient, remainder = divmod(text_width, width) 

246 except ZeroDivisionError: 

247 height = 10 ** 8 

248 else: 

249 if remainder: 

250 quotient += 1 # Like math.ceil. 

251 height = max(1, quotient) 

252 

253 # Cache and return 

254 self._line_heights_cache[key] = height 

255 return height

259 """ 

260 Control that displays formatted text. This can be either plain text, an 

261 :class:`~prompt_toolkit.formatted_text.HTML` object an 

262 :class:`~prompt_toolkit.formatted_text.ANSI` object, a list of ``(style_str, 

263 text)`` tuples or a callable that takes no argument and returns one of 

264 those, depending on how you prefer to do the formatting. See 

265 ``prompt_toolkit.layout.formatted_text`` for more information. 

266 

267 (It's mostly optimized for rather small widgets, like toolbars, menus, etc...) 

268 

269 When this UI control has the focus, the cursor will be shown in the upper 

270 left corner of this control by default. There are two ways for specifying 

271 the cursor position: 

272 

273 - Pass a `get_cursor_position` function which returns a `Point` instance 

274 with the current cursor position. 

275 

276 - If the (formatted) text is passed as a list of ``(style, text)`` tuples

278 this will specify the cursor position.

431 def mouse_handler(self, mouse_event: MouseEvent) -> "NotImplementedOrNone": 

432 """ 

433 Handle mouse events. 

434 

435 (When the fragment list contained mouse handlers and the user clicked on 

436 on any of these, the matching handler is called. This handler can still 

437 return `NotImplemented` in case we want the 

438 :class:`~prompt_toolkit.layout.Window` to handle this particular 

439 event.) 

440 """ 

441 if self._fragments: 

442 # Read the generator. 

443 fragments_for_line = list(split_lines(self._fragments)) 

444 

445 try: 

446 fragments = fragments_for_line[mouse_event.position.y] 

447 except IndexError: 

448 return NotImplemented 

449 else: 

450 # Find position in the fragment list. 

451 xpos = mouse_event.position.x 

452 

453 # Find mouse handler for this character. 

454 count = 0 

455 for item in fragments: 

456 count += len(item[1]) 

457 if count > xpos: 

458 if len(item) >= 3: 

459 # Handler found. Call it. 

460 # (Handler can return NotImplemented, so return 

461 # that result.) 

462 handler = item[2] # type: ignore 

463 return handler(mouse_event) 

464 else: 

465 break 

466 

467 # Otherwise, don't handle here. 

468 return NotImplemented

693 return i 

694 

695 transformation = merged_processor.apply_transformation( 

696 TransformationInput( 

697 self, document, lineno, source_to_display, fragments, width, height 

698 ) 

699 ) 

700 

701 return _ProcessedLine( 

702 transformation.fragments, 

703 transformation.source_to_display, 

704 transformation.display_to_source, 

705 )

707 def create_func() -> Callable[[int], _ProcessedLine]: 

708 get_line = self._get_formatted_text_for_line_func(document) 

709 cache: Dict[int, _ProcessedLine] = {} 

710 

711 def get_processed_line(i: int) -> _ProcessedLine: 

712 try: 

713 return cache[i] 

714 except KeyError: 

715 processed_line = transform(i, get_line(i)) 

716 cache[i] = processed_line 

717 return processed_line 

718 

719 return get_processed_line

721 return create_func()

723 def create_content( 

724 self, width: int, height: int, preview_search: bool = False 

725 ) -> UIContent: 

726 """ 

727 Create a UIContent. 

728 """ 

729 buffer = self.buffer 

730 

731 # Trigger history loading of the buffer. We do this during the 

732 # rendering of the UI here, because it needs to happen when an 

733 # `Application` with its event loop is running. During the rendering of 

734 # the buffer control is the earliest place we can achieve this, where 

735 # we're sure the right event loop is active, and don't require user 

736 # interaction (like in a key binding). 

737 buffer.load_history_if_not_yet_loaded() 

738 

739 # Get the document to be shown. If we are currently searching (the 

740 # search buffer has focus, and the preview_search filter is enabled), 

741 # then use the search document, which has possibly a different 

742 # text/cursor position.) 

743 search_control = self.search_buffer_control 

744 preview_now = preview_search or bool( 

745 # Only if this feature is enabled. 

746 self.preview_search() 

747 and 

748 # And something was typed in the associated search field. 

749 search_control 

750 and search_control.buffer.text 

751 and 

752 # And we are searching in this control. (Many controls can point to 

753 # the same search field, like in Pyvim.) 

754 get_app().layout.search_target_buffer_control == self 

755 ) 

756 

757 if preview_now and search_control is not None: 

758 ss = self.search_state 

759 

760 document = buffer.document_for_search( 

761 SearchState( 

762 text=search_control.buffer.text, 

763 direction=ss.direction, 

764 ignore_case=ss.ignore_case, 

765 ) 

766 ) 

767 else: 

768 document = buffer.document 

769 

770 get_processed_line = self._create_get_processed_line_func( 

771 document, width, height 

772 ) 

773 self._last_get_processed_line = get_processed_line 

774 

775 def translate_rowcol(row: int, col: int) -> Point: 

776 "Return the content column for this coordinate." 

777 return Point(x=get_processed_line(row).source_to_display(col), y=row) 

778 

779 def get_line(i: int) -> StyleAndTextTuples: 

780 "Return the fragments for a given line number." 

781 fragments = get_processed_line(i).fragments 

782 

783 # Add a space at the end, because that is a possible cursor 

784 # position. (When inserting after the input.) We should do this on 

785 # all the lines, not just the line containing the cursor. (Because 

786 # otherwise, line wrapping/scrolling could change when moving the 

787 # cursor around.) 

788 fragments = fragments + [("", " ")] 

789 return fragments 

790 

791 content = UIContent( 

792 get_line=get_line, 

793 line_count=document.line_count, 

794 cursor_position=translate_rowcol( 

795 document.cursor_position_row, document.cursor_position_col 

796 ), 

797 ) 

798 

799 # If there is an auto completion going on, use that start point for a 

800 # pop-up menu position. (But only when this buffer has the focus -- 

801 # there is only one place for a menu, determined by the focused buffer.) 

802 if get_app().layout.current_control == self: 

803 menu_position = self.menu_position() if self.menu_position else None 

804 if menu_position is not None: 

805 assert isinstance(menu_position, int) 

806 menu_row, menu_col = buffer.document.translate_index_to_position( 

807 menu_position 

808 ) 

809 content.menu_position = translate_rowcol(menu_row, menu_col) 

810 elif buffer.complete_state: 

811 # Position for completion menu. 

812 # Note: We use 'min', because the original cursor position could be 

813 # behind the input string when the actual completion is for 

814 # some reason shorter than the text we had before. (A completion 

815 # can change and shorten the input.) 

816 menu_row, menu_col = buffer.document.translate_index_to_position( 

817 min( 

818 buffer.cursor_position, 

819 buffer.complete_state.original_document.cursor_position, 

820 ) 

821 ) 

822 content.menu_position = translate_rowcol(menu_row, menu_col) 

823 else: 

824 content.menu_position = None 

825 

826 return content

828 def mouse_handler(self, mouse_event: MouseEvent) -> "NotImplementedOrNone": 

829 """ 

830 Mouse handler for this control. 

831 """ 

832 buffer = self.buffer 

833 position = mouse_event.position 

834 

835 # Focus buffer when clicked. 

836 if get_app().layout.current_control == self: 

837 if self._last_get_processed_line: 

838 processed_line = self._last_get_processed_line(position.y) 

839 

840 # Translate coordinates back to the cursor position of the 

841 # original input. 

842 xpos = processed_line.display_to_source(position.x) 

843 index = buffer.document.translate_row_col_to_index(position.y, xpos) 

844 

845 # Set the cursor position. 

846 if mouse_event.event_type == MouseEventType.MOUSE_DOWN: 

847 buffer.exit_selection() 

848 buffer.cursor_position = index 

849 

850 elif ( 

851 mouse_event.event_type == MouseEventType.MOUSE_MOVE 

852 and mouse_event.button != MouseButton.NONE 

853 ): 

854 # Click and drag to highlight a selection 

855 if ( 

856 buffer.selection_state is None 

857 and abs(buffer.cursor_position - index) > 0 

858 ): 

859 buffer.start_selection(selection_type=SelectionType.CHARACTERS) 

860 buffer.cursor_position = index 

861 

862 elif mouse_event.event_type == MouseEventType.MOUSE_UP: 

863 # When the cursor was moved to another place, select the text. 

864 # (The >1 is actually a small but acceptable workaround for 

865 # selecting text in Vi navigation mode. In navigation mode, 

866 # the cursor can never be after the text, so the cursor 

867 # will be repositioned automatically.) 

868 if abs(buffer.cursor_position - index) > 1: 

869 if buffer.selection_state is None: 

870 buffer.start_selection( 

871 selection_type=SelectionType.CHARACTERS 

872 ) 

873 buffer.cursor_position = index 

874 

875 # Select word around cursor on double click. 

876 # Two MOUSE_UP events in a short timespan are considered a double click. 

877 double_click = ( 

878 self._last_click_timestamp 

879 and time.time() - self._last_click_timestamp < 0.3 

880 ) 

881 self._last_click_timestamp = time.time() 

882 

883 if double_click: 

884 start, end = buffer.document.find_boundaries_of_current_word() 

885 buffer.cursor_position += start 

886 buffer.start_selection(selection_type=SelectionType.CHARACTERS) 

887 buffer.cursor_position += end - start 

888 else: 

889 # Don't handle scroll events here. 

890 return NotImplemented 

891 

892 # Not focused, but focusing on click events. 

893 else: 

894 if ( 

895 self.focus_on_click() 

896 and mouse_event.event_type == MouseEventType.MOUSE_UP 

897 ): 

898 # Focus happens on mouseup. (If we did this on mousedown, the 

899 # up event will be received at the point where this widget is 

900 # focused and be handled anyway.) 

901 get_app().layout.current_control = self 

902 else: 

903 return NotImplemented 

904 

905 return None
