23def get_srid_info(srid, connection): 

24 """ 

25 Return the units, unit name, and spheroid WKT associated with the 

26 given SRID from the `spatial_ref_sys` (or equivalent) spatial database 

27 table for the given database connection. These results are cached. 

28 """ 

29 from django.contrib.gis.gdal import SpatialReference 

30 global _srid_cache 

31 

32 try: 

33 # The SpatialRefSys model for the spatial backend. 

34 SpatialRefSys = connection.ops.spatial_ref_sys() 

35 except NotImplementedError: 

36 SpatialRefSys = None 

37 

38 alias, get_srs = ( 

39 (connection.alias, lambda srid: SpatialRefSys.objects.using(connection.alias).get(srid=srid).srs) 

40 if SpatialRefSys else 

41 (None, SpatialReference) 

42 ) 

43 if srid not in _srid_cache[alias]: 

44 srs = get_srs(srid) 

45 units, units_name = srs.units 

46 _srid_cache[alias][srid] = SRIDCacheEntry( 

47 units=units, 

48 units_name=units_name, 

49 spheroid='SPHEROID["%s",%s,%s]' % (srs['spheroid'], srs.semi_major, srs.inverse_flattening), 

50 geodetic=srs.geographic, 

51 ) 

52 

53 return _srid_cache[alias][srid]

155 def get_raster_prep_value(self, value, is_candidate): 

156 """ 

157 Return a GDALRaster if conversion is successful, otherwise return None. 

158 """ 

159 if isinstance(value, gdal.GDALRaster): 

160 return value 

161 elif is_candidate: 

162 try: 

163 return gdal.GDALRaster(value) 

164 except GDALException: 

165 pass 

166 elif isinstance(value, dict): 

167 try: 

168 return gdal.GDALRaster(value) 

169 except GDALException: 

170 raise ValueError("Couldn't create spatial object from lookup value '%s'." % value) 

171 

172 def get_prep_value(self, value): 

173 obj = super().get_prep_value(value) 

174 if obj is None: 

175 return None 

176 # When the input is not a geometry or raster, attempt to construct one 

177 # from the given string input. 

178 if isinstance(obj, GEOSGeometry): 

179 pass 

180 else: 

181 # Check if input is a candidate for conversion to raster or geometry. 

182 is_candidate = isinstance(obj, (bytes, str)) or hasattr(obj, '__geo_interface__') 

183 # Try to convert the input to raster. 

184 raster = self.get_raster_prep_value(obj, is_candidate) 

185 

186 if raster: 

187 obj = raster 

188 elif is_candidate: 

189 try: 

190 obj = GEOSGeometry(obj) 

191 except (GEOSException, GDALException): 

192 raise ValueError("Couldn't create spatial object from lookup value '%s'." % obj) 

193 else: 

194 raise ValueError('Cannot use object with type %s for a spatial lookup parameter.' % type(obj).__name__) 

195 

196 # Assigning the SRID value. 

197 obj.srid = self.get_srid(obj) 

198 return obj
