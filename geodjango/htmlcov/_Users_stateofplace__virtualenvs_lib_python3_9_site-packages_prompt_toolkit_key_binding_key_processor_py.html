68class KeyProcessor: 

69 """ 

70 Statemachine that receives :class:`KeyPress` instances and according to the 

71 key bindings in the given :class:`KeyBindings`, calls the matching handlers. 

72 

73 :: 

74 

75 p = KeyProcessor(key_bindings) 

76 

77 # Send keys into the processor. 

78 p.feed(KeyPress(Keys.ControlX, '\x18')) 

79 p.feed(KeyPress(Keys.ControlC, '\x03') 

80 

81 # Process all the keys in the queue. 

82 p.process_keys() 

83 

84 # Now the ControlX-ControlC callback will be called if this sequence is 

85 # registered in the key bindings. 

86 

87 :param key_bindings: `KeyBindingsBase` instance. 

88 """ 

89 

90 def __init__(self, key_bindings: KeyBindingsBase) -> None: 

91 self._bindings = key_bindings 

92 

93 self.before_key_press = Event(self) 

94 self.after_key_press = Event(self) 

95 

96 self._flush_wait_task: Optional[Task[None]] = None 

97 

98 self.reset() 

99 

100 def reset(self) -> None: 

101 self._previous_key_sequence: List[KeyPress] = [] 

102 self._previous_handler: Optional[Binding] = None 

103 

104 # The queue of keys not yet send to our _process generator/state machine. 

105 self.input_queue: Deque[KeyPress] = deque() 

106 

107 # The key buffer that is matched in the generator state machine. 

108 # (This is at at most the amount of keys that make up for one key binding.) 

109 self.key_buffer: List[KeyPress] = [] 

110 

111 #: Readline argument (for repetition of commands.) 

112 #: https://www.gnu.org/software/bash/manual/html_node/Readline-Arguments.html 

113 self.arg: Optional[str] = None 

114 

115 # Start the processor coroutine. 

116 self._process_coroutine = self._process() 

117 self._process_coroutine.send(None) # type: ignore

119 def _get_matches(self, key_presses: List[KeyPress]) -> List[Binding]: 

120 """ 

121 For a list of :class:`KeyPress` instances. Give the matching handlers 

122 that would handle this. 

123 """ 

124 keys = tuple(k.key for k in key_presses) 

125 

126 # Try match, with mode flag 

127 return [b for b in self._bindings.get_bindings_for_keys(keys) if b.filter()] 

128 

129 def _is_prefix_of_longer_match(self, key_presses: List[KeyPress]) -> bool: 

130 """ 

131 For a list of :class:`KeyPress` instances. Return True if there is any 

132 handler that is bound to a suffix of this keys. 

133 """ 

134 keys = tuple(k.key for k in key_presses) 

135 

136 # Get the filters for all the key bindings that have a longer match. 

137 # Note that we transform it into a `set`, because we don't care about 

138 # the actual bindings and executing it more than once doesn't make 

139 # sense. (Many key bindings share the same filter.) 

140 filters = set( 

141 b.filter for b in self._bindings.get_bindings_starting_with_keys(keys) 

142 ) 

143 

144 # When any key binding is active, return True. 

145 return any(f() for f in filters) 

146 

147 def _process(self) -> Generator[None, KeyPress, None]: 

148 """ 

149 Coroutine implementing the key match algorithm. Key strokes are sent 

150 into this generator, and it calls the appropriate handlers. 

151 """ 

152 buffer = self.key_buffer 

153 retry = False 

154 

155 while True: 

156 flush = False 

157 

158 if retry: 

159 retry = False 

160 else: 

161 key = yield 

162 if key is _Flush: 

163 flush = True 

164 else: 

165 buffer.append(key) 

166 

167 # If we have some key presses, check for matches. 

168 if buffer: 

169 matches = self._get_matches(buffer) 

170 

171 if flush: 

172 is_prefix_of_longer_match = False 

173 else: 

174 is_prefix_of_longer_match = self._is_prefix_of_longer_match(buffer) 

175 

176 # When eager matches were found, give priority to them and also 

177 # ignore all the longer matches. 

178 eager_matches = [m for m in matches if m.eager()] 

179 

180 if eager_matches: 

181 matches = eager_matches 

182 is_prefix_of_longer_match = False 

183 

184 # Exact matches found, call handler. 

185 if not is_prefix_of_longer_match and matches: 

186 self._call_handler(matches[-1], key_sequence=buffer[:]) 

187 del buffer[:] # Keep reference. 

188 

189 # No match found. 

190 elif not is_prefix_of_longer_match and not matches: 

191 retry = True 

192 found = False 

193 

194 # Loop over the input, try longest match first and shift. 

195 for i in range(len(buffer), 0, -1): 

196 matches = self._get_matches(buffer[:i]) 

197 if matches: 

198 self._call_handler(matches[-1], key_sequence=buffer[:i]) 

199 del buffer[:i] 

200 found = True 

201 break 

202 

203 if not found: 

204 del buffer[:1]

206 def feed(self, key_press: KeyPress, first: bool = False) -> None: 

207 """ 

208 Add a new :class:`KeyPress` to the input queue. 

209 (Don't forget to call `process_keys` in order to process the queue.) 

210 

211 :param first: If true, insert before everything else. 

212 """ 

213 if first: 

214 self.input_queue.appendleft(key_press) 

215 else: 

216 self.input_queue.append(key_press) 

217 

218 def feed_multiple(self, key_presses: List[KeyPress], first: bool = False) -> None: 

219 """ 

220 :param first: If true, insert before everything else. 

221 """ 

222 if first: 

223 self.input_queue.extendleft(reversed(key_presses)) 

224 else: 

225 self.input_queue.extend(key_presses)

227 def process_keys(self) -> None: 

228 """ 

229 Process all the keys in the `input_queue`. 

230 (To be called after `feed`.) 

231 

232 Note: because of the `feed`/`process_keys` separation, it is 

233 possible to call `feed` from inside a key binding. 

234 This function keeps looping until the queue is empty. 

235 """ 

236 app = get_app() 

237 

238 def not_empty() -> bool: 

239 # When the application result is set, stop processing keys. (E.g. 

240 # if ENTER was received, followed by a few additional key strokes, 

241 # leave the other keys in the queue.) 

242 if app.is_done: 

243 # But if there are still CPRResponse keys in the queue, these 

244 # need to be processed. 

245 return any(k for k in self.input_queue if k.key == Keys.CPRResponse) 

246 else: 

247 return bool(self.input_queue) 

248 

249 def get_next() -> KeyPress: 

250 if app.is_done: 

251 # Only process CPR responses. Everything else is typeahead. 

252 cpr = [k for k in self.input_queue if k.key == Keys.CPRResponse][0] 

253 self.input_queue.remove(cpr) 

254 return cpr 

255 else: 

256 return self.input_queue.popleft() 

257 

258 is_flush = False 

259 

260 while not_empty(): 

261 # Process next key. 

262 key_press = get_next() 

263 

264 is_flush = key_press is _Flush 

265 is_cpr = key_press.key == Keys.CPRResponse 

266 

267 if not is_flush and not is_cpr: 

268 self.before_key_press.fire() 

269 

270 try: 

271 self._process_coroutine.send(key_press) 

272 except Exception: 

273 # If for some reason something goes wrong in the parser, (maybe 

274 # an exception was raised) restart the processor for next time. 

275 self.reset() 

276 self.empty_queue() 

277 raise 

278 

279 if not is_flush and not is_cpr: 

280 self.after_key_press.fire() 

281 

282 # Skip timeout if the last key was flush. 

283 if not is_flush: 

284 self._start_timeout() 

285 

286 def empty_queue(self) -> List[KeyPress]: 

287 """ 

288 Empty the input queue. Return the unprocessed input. 

289 """ 

290 key_presses = list(self.input_queue) 

291 self.input_queue.clear() 

292 

293 # Filter out CPRs. We don't want to return these. 

294 key_presses = [k for k in key_presses if k.key != Keys.CPRResponse] 

295 return key_presses

297 def _call_handler(self, handler: Binding, key_sequence: List[KeyPress]) -> None: 

298 app = get_app() 

299 was_recording_emacs = app.emacs_state.is_recording 

300 was_recording_vi = bool(app.vi_state.recording_register) 

301 was_temporary_navigation_mode = app.vi_state.temporary_navigation_mode 

302 arg = self.arg 

303 self.arg = None 

304 

305 event = KeyPressEvent( 

306 weakref.ref(self), 

307 arg=arg, 

308 key_sequence=key_sequence, 

309 previous_key_sequence=self._previous_key_sequence, 

310 is_repeat=(handler == self._previous_handler), 

311 ) 

312 

313 # Save the state of the current buffer. 

314 if handler.save_before(event): 

315 event.app.current_buffer.save_to_undo_stack() 

316 

317 # Call handler. 

318 from prompt_toolkit.buffer import EditReadOnlyBuffer 

319 

320 try: 

321 handler.call(event) 

322 self._fix_vi_cursor_position(event) 

323 

324 except EditReadOnlyBuffer: 

325 # When a key binding does an attempt to change a buffer which is 

326 # read-only, we can ignore that. We sound a bell and go on. 

327 app.output.bell() 

328 

329 if was_temporary_navigation_mode: 

330 self._leave_vi_temp_navigation_mode(event) 

331 

332 self._previous_key_sequence = key_sequence 

333 self._previous_handler = handler 

334 

335 # Record the key sequence in our macro. (Only if we're in macro mode 

336 # before and after executing the key.) 

337 if handler.record_in_macro(): 

338 if app.emacs_state.is_recording and was_recording_emacs: 

339 recording = app.emacs_state.current_recording 

340 if recording is not None: # Should always be true, given that 

341 # `was_recording_emacs` is set. 

342 recording.extend(key_sequence) 

343 

344 if app.vi_state.recording_register and was_recording_vi: 

345 for k in key_sequence: 

346 app.vi_state.current_recording += k.data

348 def _fix_vi_cursor_position(self, event: "KeyPressEvent") -> None: 

349 """ 

350 After every command, make sure that if we are in Vi navigation mode, we 

351 never put the cursor after the last character of a line. (Unless it's 

352 an empty line.) 

353 """ 

354 app = event.app 

355 buff = app.current_buffer 

356 preferred_column = buff.preferred_column 

357 

358 if ( 

359 vi_navigation_mode() 

360 and buff.document.is_cursor_at_the_end_of_line 

361 and len(buff.document.current_line) > 0 

362 ): 

363 buff.cursor_position -= 1 

364 

365 # Set the preferred_column for arrow up/down again. 

366 # (This was cleared after changing the cursor position.) 

367 buff.preferred_column = preferred_column

369 def _leave_vi_temp_navigation_mode(self, event: "KeyPressEvent") -> None: 

370 """ 

371 If we're in Vi temporary navigation (normal) mode, return to 

372 insert/replace mode after executing one action. 

373 """ 

374 app = event.app 

375 

376 if app.editing_mode == EditingMode.VI: 

377 # Not waiting for a text object and no argument has been given. 

378 if app.vi_state.operator_func is None and self.arg is None: 

379 app.vi_state.temporary_navigation_mode = False

381 def _start_timeout(self) -> None: 

382 """ 

383 Start auto flush timeout. Similar to Vim's `timeoutlen` option. 

384 

385 Start a background coroutine with a timer. When this timeout expires 

386 and no key was pressed in the meantime, we flush all data in the queue 

387 and call the appropriate key binding handlers. 

388 """ 

389 app = get_app() 

390 timeout = app.timeoutlen 

391 

392 if timeout is None: 

393 return 

394 

395 async def wait() -> None: 

396 "Wait for timeout." 

397 # This sleep can be cancelled. In that case we don't flush. 

398 await sleep(timeout) 

399 

400 if len(self.key_buffer) > 0: 

401 # (No keys pressed in the meantime.) 

402 flush_keys() 

403 

404 def flush_keys() -> None: 

405 "Flush keys." 

406 self.feed(_Flush) 

407 self.process_keys() 

408 

409 # Automatically flush keys. 

410 if self._flush_wait_task: 

411 self._flush_wait_task.cancel() 

412 self._flush_wait_task = app.create_background_task(wait())
