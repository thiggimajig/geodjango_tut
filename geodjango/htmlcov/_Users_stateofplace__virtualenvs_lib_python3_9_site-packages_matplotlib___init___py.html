137def _parse_to_version_info(version_str): 

138 """ 

139 Parse a version string to a namedtuple analogous to sys.version_info. 

140 

141 See: 

142 https://packaging.pypa.io/en/latest/version.html#packaging.version.parse 

143 https://docs.python.org/3/library/sys.html#sys.version_info 

144 """ 

145 v = parse_version(version_str) 

146 if v.pre is None and v.post is None and v.dev is None: 

147 return _VersionInfo(v.major, v.minor, v.micro, 'final', 0) 

148 elif v.dev is not None: 

149 return _VersionInfo(v.major, v.minor, v.micro, 'alpha', v.dev) 

150 elif v.pre is not None: 

151 releaselevel = { 

152 'a': 'alpha', 

153 'b': 'beta', 

154 'rc': 'candidate'}.get(v.pre[0], 'alpha') 

155 return _VersionInfo(v.major, v.minor, v.micro, releaselevel, v.pre[1]) 

156 else: 

157 # fallback for v.post: guess-next-dev scheme from setuptools_scm 

158 return _VersionInfo(v.major, v.minor, v.micro + 1, 'alpha', v.post)

250def _logged_cached(fmt, func=None): 

251 """ 

252 Decorator that logs a function's return value, and memoizes that value. 

253 

254 After :: 

255 

256 @_logged_cached(fmt) 

257 def func(): ... 

258 

259 the first call to *func* will log its return value at the DEBUG level using 

260 %-format string *fmt*, and memoize it; later calls to *func* will directly 

261 return that value. 

262 """ 

263 if func is None: # Return the actual decorator. 

264 return functools.partial(_logged_cached, fmt) 

265 

266 called = False 

267 ret = None 

268 

269 @functools.wraps(func) 

270 def wrapper(**kwargs): 

271 nonlocal called, ret 

272 if not called: 

273 ret = func(**kwargs) 

274 called = True 

275 _log.debug(fmt, ret) 

276 return ret 

277 

278 return wrapper

292@functools.lru_cache() 

293def _get_executable_info(name): 

294 """ 

295 Get the version of some executable that Matplotlib optionally depends on. 

296 

297 .. warning:: 

298 The list of executables that this function supports is set according to 

299 Matplotlib's internal needs, and may change without notice. 

300 

301 Parameters 

302 ---------- 

303 name : str 

304 The executable to query. The following values are currently supported: 

305 "dvipng", "gs", "inkscape", "magick", "pdftops". This list is subject 

306 to change without notice. 

307 

308 Returns 

309 ------- 

310 tuple 

311 A namedtuple with fields ``executable`` (`str`) and ``version`` 

312 (`packaging.Version`, or ``None`` if the version cannot be determined). 

313 

314 Raises 

315 ------ 

316 ExecutableNotFoundError 

317 If the executable is not found or older than the oldest version 

318 supported by Matplotlib. 

319 ValueError 

320 If the executable is not one that we know how to query. 

321 """ 

322 

323 def impl(args, regex, min_ver=None, ignore_exit_code=False): 

324 # Execute the subprocess specified by args; capture stdout and stderr. 

325 # Search for a regex match in the output; if the match succeeds, the 

326 # first group of the match is the version. 

327 # Return an _ExecInfo if the executable exists, and has a version of 

328 # at least min_ver (if set); else, raise ExecutableNotFoundError. 

329 try: 

330 output = subprocess.check_output( 

331 args, stderr=subprocess.STDOUT, 

332 universal_newlines=True, errors="replace") 

333 except subprocess.CalledProcessError as _cpe: 

334 if ignore_exit_code: 

335 output = _cpe.output 

336 else: 

337 raise ExecutableNotFoundError(str(_cpe)) from _cpe 

338 except OSError as _ose: 

339 raise ExecutableNotFoundError(str(_ose)) from _ose 

340 match = re.search(regex, output) 

341 if match: 

342 version = parse_version(match.group(1)) 

343 if min_ver is not None and version < parse_version(min_ver): 

344 raise ExecutableNotFoundError( 

345 f"You have {args[0]} version {version} but the minimum " 

346 f"version supported by Matplotlib is {min_ver}") 

347 return _ExecInfo(args[0], version) 

348 else: 

349 raise ExecutableNotFoundError( 

350 f"Failed to determine the version of {args[0]} from " 

351 f"{' '.join(args)}, which output {output}") 

352 

353 if name == "dvipng": 

354 return impl(["dvipng", "-version"], "(?m)^dvipng(?: .*)? (.+)", "1.6") 

355 elif name == "gs": 

356 execs = (["gswin32c", "gswin64c", "mgs", "gs"] # "mgs" for miktex. 

357 if sys.platform == "win32" else 

358 ["gs"]) 

359 for e in execs: 

360 try: 

361 return impl([e, "--version"], "(.*)", "9") 

362 except ExecutableNotFoundError: 

363 pass 

364 message = "Failed to find a Ghostscript installation" 

365 raise ExecutableNotFoundError(message) 

366 elif name == "inkscape": 

367 try: 

368 # Try headless option first (needed for Inkscape version < 1.0): 

369 return impl(["inkscape", "--without-gui", "-V"], 

370 "Inkscape ([^ ]*)") 

371 except ExecutableNotFoundError: 

372 pass # Suppress exception chaining. 

373 # If --without-gui is not accepted, we may be using Inkscape >= 1.0 so 

374 # try without it: 

375 return impl(["inkscape", "-V"], "Inkscape ([^ ]*)") 

376 elif name == "magick": 

377 if sys.platform == "win32": 

378 # Check the registry to avoid confusing ImageMagick's convert with 

379 # Windows's builtin convert.exe. 

380 import winreg 

381 binpath = "" 

382 for flag in [0, winreg.KEY_WOW64_32KEY, winreg.KEY_WOW64_64KEY]: 

383 try: 

384 with winreg.OpenKeyEx( 

385 winreg.HKEY_LOCAL_MACHINE, 

386 r"Software\Imagemagick\Current", 

387 0, winreg.KEY_QUERY_VALUE | flag) as hkey: 

388 binpath = winreg.QueryValueEx(hkey, "BinPath")[0] 

389 except OSError: 

390 pass 

391 path = None 

392 if binpath: 

393 for name in ["convert.exe", "magick.exe"]: 

394 candidate = Path(binpath, name) 

395 if candidate.exists(): 

396 path = str(candidate) 

397 break 

398 if path is None: 

399 raise ExecutableNotFoundError( 

400 "Failed to find an ImageMagick installation") 

401 else: 

402 path = "convert" 

403 info = impl([path, "--version"], r"^Version: ImageMagick (\S*)") 

404 if info.version == parse_version("7.0.10-34"): 

405 # https://github.com/ImageMagick/ImageMagick/issues/2720 

406 raise ExecutableNotFoundError( 

407 f"You have ImageMagick {info.version}, which is unsupported") 

408 return info 

409 elif name == "pdftops": 

410 info = impl(["pdftops", "-v"], "^pdftops version (.*)", 

411 ignore_exit_code=True) 

412 if info and not ( 

413 3 <= info.version.major or 

414 # poppler version numbers. 

415 parse_version("0.9") <= info.version < parse_version("1.0")): 

416 raise ExecutableNotFoundError( 

417 f"You have pdftops version {info.version} but the minimum " 

418 f"version supported by Matplotlib is 3.0") 

419 return info 

420 else: 

421 raise ValueError("Unknown executable: {!r}".format(name))

462def _get_config_or_cache_dir(xdg_base_getter): 

463 configdir = os.environ.get('MPLCONFIGDIR') 

464 if configdir: 

465 configdir = Path(configdir).resolve() 

466 elif sys.platform.startswith(('linux', 'freebsd')): 

467 # Only call _xdg_base_getter here so that MPLCONFIGDIR is tried first, 

468 # as _xdg_base_getter can throw. 

469 configdir = Path(xdg_base_getter(), "matplotlib") 

470 else: 

471 configdir = Path.home() / ".matplotlib" 

472 try: 

473 configdir.mkdir(parents=True, exist_ok=True) 

474 except OSError: 

475 pass 

476 else: 

477 if os.access(str(configdir), os.W_OK) and configdir.is_dir(): 

478 return str(configdir) 

479 # If the config or cache directory cannot be created or is not a writable 

480 # directory, create a temporary one. 

481 tmpdir = os.environ["MPLCONFIGDIR"] = \ 

482 tempfile.mkdtemp(prefix="matplotlib-") 

483 atexit.register(shutil.rmtree, tmpdir) 

484 _log.warning( 

485 "Matplotlib created a temporary config/cache directory at %s because " 

486 "the default path (%s) is not a writable directory; it is highly " 

487 "recommended to set the MPLCONFIGDIR environment variable to a " 

488 "writable directory, in particular to speed up the import of " 

489 "Matplotlib and to better support multiprocessing.", 

490 tmpdir, configdir) 

491 return tmpdir

532 Get the location of the config file. 

533 

534 The file location is determined in the following order 

535 

536 - ``$PWD/matplotlibrc`` 

537 - ``$MATPLOTLIBRC`` if it is not a directory 

538 - ``$MATPLOTLIBRC/matplotlibrc`` 

539 - ``$MPLCONFIGDIR/matplotlibrc`` 

540 - On Linux, 

541 - ``$XDG_CONFIG_HOME/matplotlib/matplotlibrc`` (if ``$XDG_CONFIG_HOME`` 

542 is defined) 

543 - or ``$HOME/.config/matplotlib/matplotlibrc`` (if ``$XDG_CONFIG_HOME`` 

544 is not defined) 

545 - On other platforms, 

546 - ``$HOME/.matplotlib/matplotlibrc`` if ``$HOME`` is defined 

547 - Lastly, it looks in ``$MATPLOTLIBDATA/matplotlibrc``, which should always 

548 exist.

551 def gen_candidates(): 

552 # rely on down-stream code to make absolute. This protects us 

553 # from having to directly get the current working directory 

554 # which can fail if the user has ended up with a cwd that is 

555 # non-existent. 

556 yield 'matplotlibrc' 

557 try: 

558 matplotlibrc = os.environ['MATPLOTLIBRC'] 

559 except KeyError: 

560 pass 

561 else: 

562 yield matplotlibrc 

563 yield os.path.join(matplotlibrc, 'matplotlibrc') 

564 yield os.path.join(get_configdir(), 'matplotlibrc') 

565 yield os.path.join(get_data_path(), 'matplotlibrc') 

566 

567 for fname in gen_candidates(): 

568 if os.path.exists(fname) and not os.path.isdir(fname): 

569 return fname 

570 

571 raise RuntimeError("Could not find matplotlibrc file; your Matplotlib " 

572 "install is broken")

623 def __setitem__(self, key, val): 

624 try: 

625 if key in _deprecated_map: 

626 version, alt_key, alt_val, inverse_alt = _deprecated_map[key] 

627 _api.warn_deprecated( 

628 version, name=key, obj_type="rcparam", alternative=alt_key) 

629 key = alt_key 

630 val = alt_val(val) 

631 elif key in _deprecated_remain_as_none and val is not None: 

632 version, = _deprecated_remain_as_none[key] 

633 _api.warn_deprecated(version, name=key, obj_type="rcparam") 

634 elif key in _deprecated_ignore_map: 

635 version, alt_key = _deprecated_ignore_map[key] 

636 _api.warn_deprecated(

638 return

642 return 

643 try: 

644 cval = self.validate[key](val) 

645 except ValueError as ve: 

646 raise ValueError(f"Key {key}: {ve}") from None 

647 dict.__setitem__(self, key, cval) 

648 except KeyError as err: 

649 raise KeyError( 

650 f"{key} is not a valid rc parameter (see rcParams.keys() for " 

651 f"a list of valid parameters)") from err

659

759def _rc_params_in_file(fname, transform=lambda x: x, fail_on_error=False): 

760 """ 

761 Construct a `RcParams` instance from file *fname*. 

762 

763 Unlike `rc_params_from_file`, the configuration class only contains the 

764 parameters specified in the file (i.e. default values are not filled in). 

765 

766 Parameters 

767 ---------- 

768 fname : path-like 

769 The loaded file. 

770 transform : callable, default: the identity function 

771 A function called on each individual line of the file to transform it, 

772 before further parsing. 

773 fail_on_error : bool, default: False 

774 Whether invalid entries should result in an exception or a warning. 

775 """ 

776 import matplotlib as mpl 

777 rc_temp = {} 

778 with _open_file_or_url(fname) as fd: 

779 try: 

780 for line_no, line in enumerate(fd, 1): 

781 line = transform(line) 

782 strippedline = line.split('#', 1)[0].strip() 

783 if not strippedline: 

784 continue 

785 tup = strippedline.split(':', 1) 

786 if len(tup) != 2: 

787 _log.warning('Missing colon in file %r, line %d (%r)', 

788 fname, line_no, line.rstrip('\n')) 

789 continue 

790 key, val = tup 

791 key = key.strip() 

792 val = val.strip() 

793 if key in rc_temp: 

794 _log.warning('Duplicate key in file %r, line %d (%r)', 

795 fname, line_no, line.rstrip('\n')) 

796 rc_temp[key] = (val, line, line_no) 

797 except UnicodeDecodeError: 

798 _log.warning('Cannot decode configuration file %s with encoding ' 

799 '%s, check LANG and LC_* variables.', 

800 fname, 

801 locale.getpreferredencoding(do_setlocale=False) 

802 or 'utf-8 (default)') 

803 raise 

804 

805 config = RcParams() 

806 

807 for key, (val, line, line_no) in rc_temp.items(): 

808 if key in rcsetup._validators: 

809 if fail_on_error: 

810 config[key] = val # try to convert to proper type or raise 

811 else: 

812 try: 

813 config[key] = val # try to convert to proper type or skip 

814 except Exception as msg: 

815 _log.warning('Bad value in file %r, line %d (%r): %s', 

816 fname, line_no, line.rstrip('\n'), msg) 

817 elif key in _deprecated_ignore_map: 

818 version, alt_key = _deprecated_ignore_map[key] 

819 _api.warn_deprecated( 

820 version, name=key, alternative=alt_key, obj_type='rcparam', 

821 addendum="Please update your matplotlibrc.") 

822 else: 

823 # __version__ must be looked up as an attribute to trigger the 

824 # module-level __getattr__. 

825 version = ('master' if '.post' in mpl.__version__ 

826 else f'v{mpl.__version__}') 

827 _log.warning(""" 

828Bad key %(key)s in file %(fname)s, line %(line_no)s (%(line)r) 

829You probably need to get an updated matplotlibrc file from 

830https://github.com/matplotlib/matplotlib/blob/%(version)s/matplotlibrc.template 

831or from the matplotlib source distribution""", 

832 dict(key=key, fname=fname, line_no=line_no, 

833 line=line.rstrip('\n'), version=version)) 

834 return config

840 

841 Parameters 

842 ---------- 

843 fname : str or path-like 

844 A file with Matplotlib rc settings. 

845 fail_on_error : bool 

846 If True, raise an error when the parser fails to convert a parameter. 

847 use_default_template : bool 

848 If True, initialize with default parameters before updating with those 

849 in the given file. If False, the configuration class only contains the 

850 parameters specified in the file. (Useful for updating dicts.) 

851 """

855 return config_from_file

862***************************************************************** 

863You have the following UNSUPPORTED LaTeX preamble customizations: 

864%s 

865Please do not ask for support with these customizations active. 

866***************************************************************** 

867""", '\n'.join(config['text.latex.preamble']))

874# by explicitly calling the superclass (dict.update, dict.items) to avoid 

875# triggering resolution of _auto_backend_sentinel. 

876rcParamsDefault = _rc_params_in_file( 

877 cbook._get_data_path("matplotlibrc"), 

878 # Strip leading comment. 

879 transform=lambda line: line[1:] if line.startswith("#") else line, 

880 fail_on_error=True) 

881dict.update(rcParamsDefault, rcsetup._hardcoded_defaults) 

882# Normally, the default matplotlibrc file contains *no* entry for backend (the 

883# corresponding line starts with ##, not #; we fill on _auto_backend_sentinel 

884# in that case. However, packagers can set a different default backend 

885# (resulting in a normal `#backend: foo` line) in which case we should *not* 

886# fill in _auto_backend_sentinel. 

887dict.setdefault(rcParamsDefault, "backend", rcsetup._auto_backend_sentinel) 

888rcParams = RcParams() # The global instance. 

889dict.update(rcParams, dict.items(rcParamsDefault)) 

890dict.update(rcParams, _rc_params_in_file(matplotlib_fname())) 

891with _api.suppress_matplotlib_deprecation_warning(): 

892 rcParamsOrig = RcParams(rcParams.copy()) 

893 # This also checks that all rcParams are indeed listed in the template. 

894 # Assigning to rcsetup.defaultParams is left only for backcompat. 

895 defaultParams = rcsetup.defaultParams = { 

896 # We want to resolve deprecated rcParams, but not backend... 

897 key: [(rcsetup._auto_backend_sentinel if key == "backend" else 

898 rcParamsDefault[key]), 

899 validator] 

900 for key, validator in rcsetup._validators.items()} 

901if rcParams['axes.formatter.use_locale']: 

902 locale.setlocale(locale.LC_ALL, '')

905def rc(group, **kwargs): 

906 """ 

907 Set the current `.rcParams`. *group* is the grouping for the rc, e.g., 

908 for ``lines.linewidth`` the group is ``lines``, for 

909 ``axes.facecolor``, the group is ``axes``, and so on. Group may 

910 also be a list or tuple of group names, e.g., (*xtick*, *ytick*). 

911 *kwargs* is a dictionary attribute name/value pairs, e.g.,:: 

912 

913 rc('lines', linewidth=2, color='r') 

914 

915 sets the current `.rcParams` and is equivalent to:: 

916 

917 rcParams['lines.linewidth'] = 2 

918 rcParams['lines.color'] = 'r' 

919 

920 The following aliases are available to save typing for interactive users: 

921 

922 ===== ================= 

923 Alias Property 

924 ===== ================= 

925 'lw' 'linewidth' 

926 'ls' 'linestyle' 

927 'c' 'color' 

928 'fc' 'facecolor' 

929 'ec' 'edgecolor' 

930 'mew' 'markeredgewidth' 

931 'aa' 'antialiased' 

932 ===== ================= 

933 

934 Thus you could abbreviate the above call as:: 

935 

936 rc('lines', lw=2, c='r') 

937 

938 Note you can use python's kwargs dictionary facility to store 

939 dictionaries of default parameters. e.g., you can customize the 

940 font rc as follows:: 

941 

942 font = {'family' : 'monospace', 

943 'weight' : 'bold', 

944 'size' : 'larger'} 

945 rc('font', **font) # pass in the font dict as kwargs 

946 

947 This enables you to easily switch between several configurations. Use 

948 ``matplotlib.style.use('default')`` or :func:`~matplotlib.rcdefaults` to 

949 restore the default `.rcParams` after changes. 

950 

951 Notes 

952 ----- 

953 Similar functionality is available by using the normal dict interface, i.e. 

954 ``rcParams.update({"lines.linewidth": 2, ...})`` (but ``rcParams.update`` 

955 does not support abbreviations or grouping). 

956 """ 

957 

958 aliases = { 

959 'lw': 'linewidth', 

960 'ls': 'linestyle', 

961 'c': 'color', 

962 'fc': 'facecolor', 

963 'ec': 'edgecolor', 

964 'mew': 'markeredgewidth', 

965 'aa': 'antialiased', 

966 } 

967 

968 if isinstance(group, str): 

969 group = (group,) 

970 for g in group: 

971 for k, v in kwargs.items(): 

972 name = aliases.get(k) or k 

973 key = '%s.%s' % (g, name) 

974 try: 

975 rcParams[key] = v 

976 except KeyError as err: 

977 raise KeyError(('Unrecognized key "%s" for group "%s" and ' 

978 'name "%s"') % (key, g, name)) from err

1092def use(backend, *, force=True): 

1093 """ 

1094 Select the backend used for rendering and GUI integration. 

1095 

1096 Parameters 

1097 ---------- 

1098 backend : str 

1099 The backend to switch to. This can either be one of the standard 

1100 backend names, which are case-insensitive: 

1101 

1102 - interactive backends: 

1103 GTK3Agg, GTK3Cairo, GTK4Agg, GTK4Cairo, MacOSX, nbAgg, QtAgg, 

1104 QtCairo, TkAgg, TkCairo, WebAgg, WX, WXAgg, WXCairo, Qt5Agg, Qt5Cairo 

1105 

1106 - non-interactive backends: 

1107 agg, cairo, pdf, pgf, ps, svg, template 

1108 

1109 or a string of the form: ``module://my.module.name``. 

1110 

1111 Switching to an interactive backend is not possible if an unrelated 

1112 event loop has already been started (e.g., switching to GTK3Agg if a 

1113 TkAgg window has already been opened). Switching to a non-interactive 

1114 backend is always possible. 

1115 

1116 force : bool, default: True 

1117 If True (the default), raise an `ImportError` if the backend cannot be 

1118 set up (either because it fails to import, or because an incompatible 

1119 GUI interactive framework is already running); if False, silently 

1120 ignore the failure. 

1121 

1122 See Also 

1123 -------- 

1124 :ref:`backends` 

1125 matplotlib.get_backend 

1126 """ 

1127 name = validate_backend(backend) 

1128 # we need to use the base-class method here to avoid (prematurely) 

1129 # resolving the "auto" backend setting 

1130 if dict.__getitem__(rcParams, 'backend') == name: 

1131 # Nothing to do if the requested backend is already set 

1132 pass 

1133 else: 

1134 # if pyplot is not already imported, do not import it. Doing 

1135 # so may trigger a `plt.switch_backend` to the _default_ backend 

1136 # before we get a chance to change to the one the user just requested 

1137 plt = sys.modules.get('matplotlib.pyplot') 

1138 # if pyplot is imported, then try to change backends 

1139 if plt is not None: 

1140 try: 

1141 # we need this import check here to re-raise if the 

1142 # user does not have the libraries to support their 

1143 # chosen backend installed. 

1144 plt.switch_backend(name) 

1145 except ImportError: 

1146 if force: 

1147 raise 

1148 # if we have not imported pyplot, then we can set the rcParam 

1149 # value which will be respected when the user finally imports 

1150 # pyplot 

1151 else: 

1152 rcParams['backend'] = backend 

1153 # if the user has asked for a given backend, do not helpfully 

1154 # fallback 

1155 rcParams['backend_fallback'] = False 

1156 

1157 

1158if os.environ.get('MPLBACKEND'): 

1159 rcParams['backend'] = os.environ.get('MPLBACKEND')

1302 replace_names : list of str or None 

1303 The list of parameter names which arguments should be replaced by 

1304 ``data[name]`` (if ``data[name]`` does not throw an exception). If 

1305 None, replacement is attempted for all arguments. 

1306 

1307 Returns 

1308 ------- 

1309 str 

1310 The augmented docstring. 

1311 """

1323 

1324 {', '.join(map('*{}*'.format, replace_names))}""")

1338def _preprocess_data(func=None, *, replace_names=None, label_namer=None): 

1339 """ 

1340 A decorator to add a 'data' kwarg to a function. 

1341 

1342 When applied:: 

1343 

1344 @_preprocess_data() 

1345 def func(ax, *args, **kwargs): ... 

1346 

1347 the signature is modified to ``decorated(ax, *args, data=None, **kwargs)`` 

1348 with the following behavior: 

1349 

1350 - if called with ``data=None``, forward the other arguments to ``func``; 

1351 - otherwise, *data* must be a mapping; for any argument passed in as a 

1352 string ``name``, replace the argument by ``data[name]`` (if this does not 

1353 throw an exception), then forward the arguments to ``func``. 

1354 

1355 In either case, any argument that is a `MappingView` is also converted to a 

1356 list. 

1357 

1358 Parameters 

1359 ---------- 

1360 replace_names : list of str or None, default: None 

1361 The list of parameter names for which lookup into *data* should be 

1362 attempted. If None, replacement is attempted for all arguments. 

1363 label_namer : str, default: None 

1364 If set e.g. to "namer" (which must be a kwarg in the function's 

1365 signature -- not as ``**kwargs``), if the *namer* argument passed in is 

1366 a (string) key of *data* and no *label* kwarg is passed, then use the 

1367 (string) value of the *namer* as *label*. :: 

1368 

1369 @_preprocess_data(label_namer="foo") 

1370 def func(foo, label=None): ... 

1371 

1372 func("key", data={"key": value}) 

1373 # is equivalent to 

1374 func.__wrapped__(value, label="key") 

1375 """ 

1376 

1377 if func is None: # Return the actual decorator. 

1378 return functools.partial( 

1379 _preprocess_data, 

1380 replace_names=replace_names, label_namer=label_namer) 

1381 

1382 sig = inspect.signature(func) 

1383 varargs_name = None 

1384 varkwargs_name = None 

1385 arg_names = [] 

1386 params = list(sig.parameters.values()) 

1387 for p in params: 

1388 if p.kind is Parameter.VAR_POSITIONAL: 

1389 varargs_name = p.name 

1390 elif p.kind is Parameter.VAR_KEYWORD: 

1391 varkwargs_name = p.name 

1392 else: 

1393 arg_names.append(p.name) 

1394 data_param = Parameter("data", Parameter.KEYWORD_ONLY, default=None) 

1395 if varkwargs_name: 

1396 params.insert(-1, data_param) 

1397 else: 

1398 params.append(data_param) 

1399 new_sig = sig.replace(parameters=params) 

1400 arg_names = arg_names[1:] # remove the first "ax" / self arg 

1401 

1402 assert {*arg_names}.issuperset(replace_names or []) or varkwargs_name, ( 

1403 "Matplotlib internal error: invalid replace_names ({!r}) for {!r}" 

1404 .format(replace_names, func.__name__)) 

1405 assert label_namer is None or label_namer in arg_names, ( 

1406 "Matplotlib internal error: invalid label_namer ({!r}) for {!r}" 

1407 .format(label_namer, func.__name__)) 

1408 

1409 @functools.wraps(func) 

1410 def inner(ax, *args, data=None, **kwargs): 

1411 if data is None: 

1412 return func(ax, *map(sanitize_sequence, args), **kwargs) 

1413 

1414 bound = new_sig.bind(ax, *args, **kwargs) 

1415 auto_label = (bound.arguments.get(label_namer) 

1416 or bound.kwargs.get(label_namer)) 

1417 

1418 for k, v in bound.arguments.items(): 

1419 if k == varkwargs_name: 

1420 for k1, v1 in v.items(): 

1421 if replace_names is None or k1 in replace_names: 

1422 v[k1] = _replacer(data, v1) 

1423 elif k == varargs_name: 

1424 if replace_names is None: 

1425 bound.arguments[k] = tuple(_replacer(data, v1) for v1 in v) 

1426 else: 

1427 if replace_names is None or k in replace_names: 

1428 bound.arguments[k] = _replacer(data, v) 

1429 

1430 new_args = bound.args 

1431 new_kwargs = bound.kwargs 

1432 

1433 args_and_kwargs = {**bound.arguments, **bound.kwargs} 

1434 if label_namer and "label" not in args_and_kwargs: 

1435 new_kwargs["label"] = _label_from_arg( 

1436 args_and_kwargs.get(label_namer), auto_label) 

1437 

1438 return func(*new_args, **new_kwargs) 

1439 

1440 inner.__doc__ = _add_data_doc(inner.__doc__, replace_names) 

1441 inner.__signature__ = new_sig 

1442 return inner
