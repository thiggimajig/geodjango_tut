33def cast_to_common_type(arr: ArrayLike, dtype: DtypeObj) -> ArrayLike: 

34 """ 

35 Helper function for `arr.astype(common_dtype)` but handling all special 

36 cases. 

37 """ 

38 if is_dtype_equal(arr.dtype, dtype): 

39 return arr 

40 if ( 

41 is_categorical_dtype(arr.dtype) 

42 and isinstance(dtype, np.dtype) 

43 and np.issubdtype(dtype, np.integer) 

44 ): 

45 # problem case: categorical of int -> gives int as result dtype, 

46 # but categorical can contain NAs -> fall back to object dtype 

47 try: 

48 return arr.astype(dtype, copy=False) 

49 except ValueError: 

50 return arr.astype(object, copy=False) 

51 

52 if is_sparse(arr) and not is_sparse(dtype): 

53 # problem case: SparseArray.astype(dtype) doesn't follow the specified 

54 # dtype exactly, but converts this to Sparse[dtype] -> first manually 

55 # convert to dense array 

56 arr = cast(SparseArray, arr) 

57 return arr.to_dense().astype(dtype, copy=False) 

58 

59 if ( 

60 isinstance(arr, np.ndarray) 

61 and arr.dtype.kind in ["m", "M"] 

62 and dtype is np.dtype("object") 

63 ): 

64 # wrap datetime-likes in EA to ensure astype(object) gives Timestamp/Timedelta 

65 # this can happen when concat_compat is called directly on arrays (when arrays 

66 # are not coming from Index/Series._values), eg in BlockManager.quantile 

67 arr = ensure_wrapped_if_datetimelike(arr) 

68 

69 if isinstance(dtype, ExtensionDtype): 

70 if isinstance(arr, np.ndarray): 

71 # numpy's astype cannot handle ExtensionDtypes 

72 return pd_array(arr, dtype=dtype, copy=False) 

73 return arr.astype(dtype, copy=False) 

74 

75 return arr.astype(dtype, copy=False)

78def concat_compat(to_concat, axis: int = 0, ea_compat_axis: bool = False): 

79 """ 

80 provide concatenation of an array of arrays each of which is a single 

81 'normalized' dtypes (in that for example, if it's object, then it is a 

82 non-datetimelike and provide a combined dtype for the resulting array that 

83 preserves the overall dtype if possible) 

84 

85 Parameters 

86 ---------- 

87 to_concat : array of arrays 

88 axis : axis to provide concatenation 

89 ea_compat_axis : bool, default False 

90 For ExtensionArray compat, behave as if axis == 1 when determining 

91 whether to drop empty arrays. 

92 

93 Returns 

94 ------- 

95 a single array, preserving the combined dtypes 

96 """ 

97 # filter empty arrays 

98 # 1-d dtypes always are included here 

99 def is_nonempty(x) -> bool: 

100 if x.ndim <= axis: 

101 return True 

102 return x.shape[axis] > 0 

103 

104 # If all arrays are empty, there's nothing to convert, just short-cut to 

105 # the concatenation, #3121. 

106 # 

107 # Creating an empty array directly is tempting, but the winnings would be 

108 # marginal given that it would still require shape & dtype calculation and 

109 # np.concatenate which has them both implemented is compiled. 

110 non_empties = [x for x in to_concat if is_nonempty(x)] 

111 if non_empties and axis == 0 and not ea_compat_axis: 

112 # ea_compat_axis see GH#39574 

113 to_concat = non_empties 

114 

115 kinds = {obj.dtype.kind for obj in to_concat} 

116 contains_datetime = any(kind in ["m", "M"] for kind in kinds) 

117 

118 all_empty = not len(non_empties) 

119 single_dtype = len({x.dtype for x in to_concat}) == 1 

120 any_ea = any(isinstance(x.dtype, ExtensionDtype) for x in to_concat) 

121 

122 if contains_datetime: 

123 return _concat_datetime(to_concat, axis=axis) 

124 

125 if any_ea: 

126 # we ignore axis here, as internally concatting with EAs is always 

127 # for axis=0 

128 if not single_dtype: 

129 target_dtype = find_common_type([x.dtype for x in to_concat]) 

130 to_concat = [cast_to_common_type(arr, target_dtype) for arr in to_concat] 

131 

132 if isinstance(to_concat[0], ExtensionArray): 

133 cls = type(to_concat[0]) 

134 return cls._concat_same_type(to_concat) 

135 else: 

136 return np.concatenate(to_concat) 

137 

138 elif all_empty: 

139 # we have all empties, but may need to coerce the result dtype to 

140 # object if we have non-numeric type operands (numpy would otherwise 

141 # cast this to float) 

142 if len(kinds) != 1: 

143 

144 if not len(kinds - {"i", "u", "f"}) or not len(kinds - {"b", "i", "u"}): 

145 # let numpy coerce 

146 pass 

147 else: 

148 # coerce to object 

149 to_concat = [x.astype("object") for x in to_concat] 

150 

151 return np.concatenate(to_concat, axis=axis)

154def union_categoricals( 

155 to_union, sort_categories: bool = False, ignore_order: bool = False 

156): 

157 """ 

158 Combine list-like of Categorical-like, unioning categories. 

159 

160 All categories must have the same dtype. 

161 

162 Parameters 

163 ---------- 

164 to_union : list-like 

165 Categorical, CategoricalIndex, or Series with dtype='category'. 

166 sort_categories : bool, default False 

167 If true, resulting categories will be lexsorted, otherwise 

168 they will be ordered as they appear in the data. 

169 ignore_order : bool, default False 

170 If true, the ordered attribute of the Categoricals will be ignored. 

171 Results in an unordered categorical. 

172 

173 Returns 

174 ------- 

175 Categorical 

176 

177 Raises 

178 ------ 

179 TypeError 

180 - all inputs do not have the same dtype 

181 - all inputs do not have the same ordered property 

182 - all inputs are ordered and their categories are not identical 

183 - sort_categories=True and Categoricals are ordered 

184 ValueError 

185 Empty list of categoricals passed 

186 

187 Notes 

188 ----- 

189 To learn more about categories, see `link 

190 <https://pandas.pydata.org/pandas-docs/stable/user_guide/categorical.html#unioning>`__ 

191 

192 Examples 

193 -------- 

194 >>> from pandas.api.types import union_categoricals 

195 

196 If you want to combine categoricals that do not necessarily have 

197 the same categories, `union_categoricals` will combine a list-like 

198 of categoricals. The new categories will be the union of the 

199 categories being combined. 

200 

201 >>> a = pd.Categorical(["b", "c"]) 

202 >>> b = pd.Categorical(["a", "b"]) 

203 >>> union_categoricals([a, b]) 

204 ['b', 'c', 'a', 'b'] 

205 Categories (3, object): ['b', 'c', 'a'] 

206 

207 By default, the resulting categories will be ordered as they appear 

208 in the `categories` of the data. If you want the categories to be 

209 lexsorted, use `sort_categories=True` argument. 

210 

211 >>> union_categoricals([a, b], sort_categories=True) 

212 ['b', 'c', 'a', 'b'] 

213 Categories (3, object): ['a', 'b', 'c'] 

214 

215 `union_categoricals` also works with the case of combining two 

216 categoricals of the same categories and order information (e.g. what 

217 you could also `append` for). 

218 

219 >>> a = pd.Categorical(["a", "b"], ordered=True) 

220 >>> b = pd.Categorical(["a", "b", "a"], ordered=True) 

221 >>> union_categoricals([a, b]) 

222 ['a', 'b', 'a', 'b', 'a'] 

223 Categories (2, object): ['a' < 'b'] 

224 

225 Raises `TypeError` because the categories are ordered and not identical. 

226 

227 >>> a = pd.Categorical(["a", "b"], ordered=True) 

228 >>> b = pd.Categorical(["a", "b", "c"], ordered=True) 

229 >>> union_categoricals([a, b]) 

230 Traceback (most recent call last): 

231 ... 

232 TypeError: to union ordered Categoricals, all categories must be the same 

233 

234 New in version 0.20.0 

235 

236 Ordered categoricals with different categories or orderings can be 

237 combined by using the `ignore_ordered=True` argument. 

238 

239 >>> a = pd.Categorical(["a", "b", "c"], ordered=True) 

240 >>> b = pd.Categorical(["c", "b", "a"], ordered=True) 

241 >>> union_categoricals([a, b], ignore_order=True) 

242 ['a', 'b', 'c', 'c', 'b', 'a'] 

243 Categories (3, object): ['a', 'b', 'c'] 

244 

245 `union_categoricals` also works with a `CategoricalIndex`, or `Series` 

246 containing categorical data, but note that the resulting array will 

247 always be a plain `Categorical` 

248 

249 >>> a = pd.Series(["b", "c"], dtype='category') 

250 >>> b = pd.Series(["a", "b"], dtype='category') 

251 >>> union_categoricals([a, b]) 

252 ['b', 'c', 'a', 'b'] 

253 Categories (3, object): ['b', 'c', 'a'] 

254 """ 

255 from pandas import Categorical 

256 from pandas.core.arrays.categorical import recode_for_categories 

257 

258 if len(to_union) == 0: 

259 raise ValueError("No Categoricals to union") 

260 

261 def _maybe_unwrap(x): 

262 if isinstance(x, (ABCCategoricalIndex, ABCSeries)): 

263 return x._values 

264 elif isinstance(x, Categorical): 

265 return x 

266 else: 

267 raise TypeError("all components to combine must be Categorical") 

268 

269 to_union = [_maybe_unwrap(x) for x in to_union] 

270 first = to_union[0] 

271 

272 if not all( 

273 is_dtype_equal(other.categories.dtype, first.categories.dtype) 

274 for other in to_union[1:] 

275 ): 

276 raise TypeError("dtype of categories must be the same") 

277 

278 ordered = False 

279 if all(first._categories_match_up_to_permutation(other) for other in to_union[1:]): 

280 # identical categories - fastpath 

281 categories = first.categories 

282 ordered = first.ordered 

283 

284 all_codes = [first._encode_with_my_categories(x)._codes for x in to_union] 

285 new_codes = np.concatenate(all_codes) 

286 

287 if sort_categories and not ignore_order and ordered: 

288 raise TypeError("Cannot use sort_categories=True with ordered Categoricals") 

289 

290 if sort_categories and not categories.is_monotonic_increasing: 

291 categories = categories.sort_values() 

292 indexer = categories.get_indexer(first.categories) 

293 

294 from pandas.core.algorithms import take_nd 

295 

296 new_codes = take_nd(indexer, new_codes, fill_value=-1) 

297 elif ignore_order or all(not c.ordered for c in to_union): 

298 # different categories - union and recode 

299 cats = first.categories.append([c.categories for c in to_union[1:]]) 

300 categories = cats.unique() 

301 if sort_categories: 

302 categories = categories.sort_values() 

303 

304 new_codes = [ 

305 recode_for_categories(c.codes, c.categories, categories) for c in to_union 

306 ] 

307 new_codes = np.concatenate(new_codes) 

308 else: 

309 # ordered - to show a proper error message 

310 if all(c.ordered for c in to_union): 

311 msg = "to union ordered Categoricals, all categories must be the same" 

312 raise TypeError(msg) 

313 else: 

314 raise TypeError("Categorical.ordered must be the same") 

315 

316 if ignore_order: 

317 ordered = False 

318 

319 return Categorical(new_codes, categories=categories, ordered=ordered, fastpath=True)
