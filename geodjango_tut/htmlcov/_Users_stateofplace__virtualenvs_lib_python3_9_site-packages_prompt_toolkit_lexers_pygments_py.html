39class SyntaxSync(metaclass=ABCMeta): 

40 """ 

41 Syntax synchroniser. This is a tool that finds a start position for the 

42 lexer. This is especially important when editing big documents; we don't 

43 want to start the highlighting by running the lexer from the beginning of 

44 the file. That is very slow when editing. 

45 """ 

46

48 def get_sync_start_position( 

49 self, document: Document, lineno: int

51 """ 

52 Return the position from where we can start lexing as a (row, column) 

53 tuple. 

54 

55 :param document: `Document` instance that contains all the lines. 

56 :param lineno: The line that we want to highlight. (We need to return 

57 this line, or an earlier position.) 

58 """

72class RegexSync(SyntaxSync): 

73 """ 

74 Synchronize by starting at a line that matches the given regex pattern. 

75 """ 

76 

77 # Never go more than this amount of lines backwards for synchronisation. 

78 # That would be too CPU intensive. 

79 MAX_BACKWARDS = 500 

80 

81 # Start lexing at the start, if we are in the first 'n' lines and no 

82 # synchronisation position was found. 

83 FROM_START_IF_NO_SYNC_POS_FOUND = 100 

84 

85 def __init__(self, pattern: str) -> None: 

86 self._compiled_pattern = re.compile(pattern) 

87 

88 def get_sync_start_position( 

89 self, document: Document, lineno: int 

90 ) -> Tuple[int, int]: 

91 """ 

92 Scan backwards, and find a possible position to start. 

93 """ 

94 pattern = self._compiled_pattern 

95 lines = document.lines 

96 

97 # Scan upwards, until we find a point where we can start the syntax 

98 # synchronisation. 

99 for i in range(lineno, max(-1, lineno - self.MAX_BACKWARDS), -1): 

100 match = pattern.match(lines[i]) 

101 if match: 

102 return i, match.start() 

103 

104 # No synchronisation point found. If we aren't that far from the 

105 # beginning, start at the very beginning, otherwise, just try to start 

106 # at the current line. 

107 if lineno < self.FROM_START_IF_NO_SYNC_POS_FOUND: 

108 return 0, 0 

109 else: 

110 return lineno, 0

112 @classmethod 

113 def from_pygments_lexer_cls(cls, lexer_cls: "PygmentsLexerCls") -> "RegexSync": 

114 """ 

115 Create a :class:`.RegexSync` instance for this Pygments lexer class. 

116 """ 

117 patterns = { 

118 # For Python, start highlighting at any class/def block. 

119 "Python": r"^\s*(class|def)\s+", 

120 "Python 3": r"^\s*(class|def)\s+", 

121 # For HTML, start at any open/close tag definition. 

122 "HTML": r"<[/a-zA-Z]", 

123 # For javascript, start at a function. 

124 "JavaScript": r"\bfunction\b" 

125 # TODO: Add definitions for other languages. 

126 # By default, we start at every possible line. 

127 } 

128 p = patterns.get(lexer_cls.name, "^") 

129 return cls(p)

149class PygmentsLexer(Lexer): 

150 """ 

151 Lexer that calls a pygments lexer. 

152 

153 Example:: 

154 

155 from pygments.lexers.html import HtmlLexer 

156 lexer = PygmentsLexer(HtmlLexer) 

157 

158 Note: Don't forget to also load a Pygments compatible style. E.g.:: 

159 

160 from prompt_toolkit.styles.from_pygments import style_from_pygments_cls 

161 from pygments.styles import get_style_by_name 

162 style = style_from_pygments_cls(get_style_by_name('monokai')) 

163 

164 :param pygments_lexer_cls: A `Lexer` from Pygments. 

165 :param sync_from_start: Start lexing at the start of the document. This 

166 will always give the best results, but it will be slow for bigger 

167 documents. (When the last part of the document is display, then the 

168 whole document will be lexed by Pygments on every key stroke.) It is 

169 recommended to disable this for inputs that are expected to be more 

170 than 1,000 lines. 

171 :param syntax_sync: `SyntaxSync` object. 

172 """ 

173 

174 # Minimum amount of lines to go backwards when starting the parser. 

175 # This is important when the lines are retrieved in reverse order, or when 

176 # scrolling upwards. (Due to the complexity of calculating the vertical 

177 # scroll offset in the `Window` class, lines are not always retrieved in 

178 # order.) 

179 MIN_LINES_BACKWARDS = 50 

180 

181 # When a parser was started this amount of lines back, read the parser 

182 # until we get the current line. Otherwise, start a new parser. 

183 # (This should probably be bigger than MIN_LINES_BACKWARDS.) 

184 REUSE_GENERATOR_MAX_DISTANCE = 100 

185 

186 def __init__( 

187 self, 

188 pygments_lexer_cls: Type["PygmentsLexerCls"], 

189 sync_from_start: FilterOrBool = True, 

190 syntax_sync: Optional[SyntaxSync] = None, 

191 ) -> None: 

192 

193 self.pygments_lexer_cls = pygments_lexer_cls 

194 self.sync_from_start = to_filter(sync_from_start) 

195 

196 # Instantiate the Pygments lexer. 

197 self.pygments_lexer = pygments_lexer_cls( 

198 stripnl=False, stripall=False, ensurenl=False 

199 ) 

200 

201 # Create syntax sync instance. 

202 self.syntax_sync = syntax_sync or RegexSync.from_pygments_lexer_cls( 

203 pygments_lexer_cls 

204 )

226 Create a lexer function that takes a line number and returns the list

245 def find_closest_generator(i: int) -> Optional[LineGenerator]: 

246 "Return a generator close to line 'i', or None if none was found." 

247 for generator, lineno in line_generators.items(): 

248 if lineno < i and i - lineno < self.REUSE_GENERATOR_MAX_DISTANCE:

252 def create_line_generator(start_lineno: int, column: int = 0) -> LineGenerator: 

253 """ 

254 Create a generator that yields the lexed lines. 

255 Each iteration it yields a (line_number, [(style_str, text), ...]) tuple. 

256 """ 

257 

258 def get_text_fragments() -> Iterable[Tuple[str, str]]: 

259 text = "\n".join(document.lines[start_lineno:])[column:] 

260 

261 # We call `get_text_fragments_unprocessed`, because `get_tokens` will 

262 # still replace \r\n and \r by \n. (We don't want that, 

263 # Pygments should return exactly the same amount of text, as we 

264 # have given as input.) 

265 for _, t, v in self.pygments_lexer.get_tokens_unprocessed(text): 

266 # Turn Pygments `Token` object into prompt_toolkit style 

267 # strings. 

268 yield _token_cache[t], v 

269 

270 yield from enumerate(split_lines(list(get_text_fragments())), start_lineno)

272 def get_generator(i: int) -> LineGenerator: 

273 """ 

274 Find an already started generator that is close, or create a new one. 

275 """ 

276 # Find closest line generator. 

277 generator = find_closest_generator(i) 

278 if generator: 

279 return generator 

280 

281 # No generator found. Determine starting point for the syntax 

282 # synchronisation first. 

283 

284 # Go at least x lines back. (Make scrolling upwards more 

285 # efficient.) 

286 i = max(0, i - self.MIN_LINES_BACKWARDS) 

287 

288 if i == 0: 

289 row = 0 

290 column = 0 

291 else: 

292 row, column = get_syntax_sync().get_sync_start_position(document, i) 

293 

294 # Find generator close to this point, or otherwise create a new one. 

295 generator = find_closest_generator(i) 

296 if generator: 

297 return generator 

298 else: 

299 generator = create_line_generator(row, column) 

300 

301 # If the column is not 0, ignore the first line. (Which is 

302 # incomplete. This happens when the synchronisation algorithm tells 

303 # us to start parsing in the middle of a line.) 

304 if column: 

305 next(generator) 

306 row += 1 

307 

308 line_generators[generator] = row 

309 return generator 

310 

311 def get_line(i: int) -> StyleAndTextTuples: 

312 "Return the tokens for a given line number." 

313 try: 

314 return cache[i] 

315 except KeyError: 

316 generator = get_generator(i) 

317 

318 # Exhaust the generator, until we find the requested line. 

319 for num, line in generator: 

320 cache[num] = line 

321 if num == i: 

322 line_generators[generator] = i 

323 

324 # Remove the next item from the cache. 

325 # (It could happen that it's already there, because of 

326 # another generator that started filling these lines, 

327 # but we want to synchronise these lines with the 

328 # current lexer's state.) 

329 if num + 1 in cache: 

330 del cache[num + 1] 

331 

332 return cache[num] 

333 return [] 

334 

335 return get_line
