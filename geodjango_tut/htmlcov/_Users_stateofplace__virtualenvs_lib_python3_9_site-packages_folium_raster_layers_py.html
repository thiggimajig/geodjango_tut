19class TileLayer(Layer): 

20 """ 

21 Create a tile layer to append on a Map. 

22 

23 Parameters 

24 ---------- 

25 tiles: str, default 'OpenStreetMap' 

26 Map tileset to use. Can choose from this list of built-in tiles: 

27 - "OpenStreetMap" 

28 - "Stamen Terrain", "Stamen Toner", "Stamen Watercolor" 

29 - "CartoDB positron", "CartoDB dark_matter" 

30 

31 You can pass a custom tileset to Folium by passing a Leaflet-style 

32 URL to the tiles parameter: ``http://{s}.yourtiles.com/{z}/{x}/{y}.png``. 

33 

34 You can find a list of free tile providers here: 

35 ``http://leaflet-extras.github.io/leaflet-providers/preview/``. 

36 Be sure to check their terms and conditions and to provide attribution 

37 with the `attr` keyword. 

38 min_zoom: int, default 0 

39 Minimum allowed zoom level for this tile layer. 

40 max_zoom: int, default 18 

41 Maximum allowed zoom level for this tile layer. 

42 max_native_zoom: int, default None 

43 The highest zoom level at which the tile server can provide tiles. 

44 If provided you can zoom in past this level. Else tiles will turn grey. 

45 attr: string, default None 

46 Map tile attribution; only required if passing custom tile URL. 

47 detect_retina: bool, default False 

48 If true and user is on a retina display, it will request four 

49 tiles of half the specified size and a bigger zoom level in place 

50 of one to utilize the high resolution. 

51 name : string, default None 

52 The name of the Layer, as it will appear in LayerControls 

53 overlay : bool, default False 

54 Adds the layer as an optional overlay (True) or the base layer (False). 

55 control : bool, default True 

56 Whether the Layer will be included in LayerControls. 

57 show: bool, default True 

58 Whether the layer will be shown on opening (only for overlays). 

59 subdomains: list of strings, default ['abc'] 

60 Subdomains of the tile service. 

61 tms: bool, default False 

62 If true, inverses Y axis numbering for tiles (turn this on for TMS 

63 services). 

64 opacity: float, default 1 

65 Sets the opacity for the layer. 

66 **kwargs : additional keyword arguments 

67 Other keyword arguments are passed as options to the Leaflet tileLayer 

68 object. 

69 """ 

70 _template = Template(u""" 

71 {% macro script(this, kwargs) %} 

72 var {{ this.get_name() }} = L.tileLayer( 

73 {{ this.tiles|tojson }}, 

74 {{ this.options|tojson }} 

75 ).addTo({{ this._parent.get_name() }}); 

76 {% endmacro %} 

77 """)

79 def __init__(self, tiles='OpenStreetMap', min_zoom=0, max_zoom=18, 

80 max_native_zoom=None, attr=None, 

81 detect_retina=False, name=None, overlay=False, 

82 control=True, show=True, no_wrap=False, subdomains='abc', 

83 tms=False, opacity=1, **kwargs): 

84 

85 self.tile_name = (name if name is not None else 

86 ''.join(tiles.lower().strip().split())) 

87 super(TileLayer, self).__init__(name=self.tile_name, overlay=overlay, 

88 control=control, show=show) 

89 self._name = 'TileLayer' 

90 self._env = ENV 

91 

92 tiles_flat = ''.join(tiles.lower().strip().split()) 

93 if tiles_flat in {'cloudmade', 'mapbox', 'mapboxbright', 'mapboxcontrolroom'}: 

94 # added in May 2020 after v0.11.0, remove in a future release 

95 raise ValueError( 

96 'Built-in templates for Mapbox and Cloudmade have been removed. ' 

97 'You can still use these providers by passing a URL to the `tiles` ' 

98 'argument. See the documentation of the `TileLayer` class.' 

99 ) 

100 templates = list(self._env.list_templates( 

101 filter_func=lambda x: x.startswith('tiles/'))) 

102 tile_template = 'tiles/' + tiles_flat + '/tiles.txt' 

103 attr_template = 'tiles/' + tiles_flat + '/attr.txt' 

104 

105 if tile_template in templates and attr_template in templates: 

106 self.tiles = self._env.get_template(tile_template).render() 

107 attr = self._env.get_template(attr_template).render() 

108 else: 

109 self.tiles = tiles 

110 if not attr: 

111 raise ValueError('Custom tiles must have an attribution.') 

112 

113 self.options = parse_options( 

114 min_zoom=min_zoom, 

115 max_zoom=max_zoom, 

116 max_native_zoom=max_native_zoom or max_zoom, 

117 no_wrap=no_wrap, 

118 attribution=attr, 

119 subdomains=subdomains, 

120 detect_retina=detect_retina, 

121 tms=tms, 

122 opacity=opacity, 

123 **kwargs 

124 )

189class ImageOverlay(Layer): 

190 """ 

191 Used to load and display a single image over specific bounds of 

192 the map, implements ILayer interface. 

193 

194 Parameters 

195 ---------- 

196 image: string, file or array-like object 

197 The data you want to draw on the map. 

198 * If string, it will be written directly in the output file. 

199 * If file, it's content will be converted as embedded in the output file. 

200 * If array-like, it will be converted to PNG base64 string and embedded in the output. 

201 bounds: list 

202 Image bounds on the map in the form 

203 [[lat_min, lon_min], [lat_max, lon_max]] 

204 opacity: float, default Leaflet's default (1.0) 

205 alt: string, default Leaflet's default ('') 

206 origin: ['upper' | 'lower'], optional, default 'upper' 

207 Place the [0,0] index of the array in the upper left or 

208 lower left corner of the axes. 

209 colormap: callable, used only for `mono` image. 

210 Function of the form [x -> (r,g,b)] or [x -> (r,g,b,a)] 

211 for transforming a mono image into RGB. 

212 It must output iterables of length 3 or 4, 

213 with values between 0 and 1. 

214 Hint: you can use colormaps from `matplotlib.cm`. 

215 mercator_project: bool, default False. 

216 Used only for array-like image. Transforms the data to 

217 project (longitude, latitude) coordinates to the Mercator projection. 

218 Beware that this will only work if `image` is an array-like object. 

219 pixelated: bool, default True 

220 Sharp sharp/crips (True) or aliased corners (False). 

221 name : string, default None 

222 The name of the Layer, as it will appear in LayerControls 

223 overlay : bool, default True 

224 Adds the layer as an optional overlay (True) or the base layer (False). 

225 control : bool, default True 

226 Whether the Layer will be included in LayerControls. 

227 show: bool, default True 

228 Whether the layer will be shown on opening (only for overlays). 

229 

230 See https://leafletjs.com/reference-1.6.0.html#imageoverlay for more 

231 options. 

232 

233 """ 

234 _template = Template(u""" 

235 {% macro script(this, kwargs) %} 

236 var {{ this.get_name() }} = L.imageOverlay( 

237 {{ this.url|tojson }}, 

238 {{ this.bounds|tojson }}, 

239 {{ this.options|tojson }} 

240 ).addTo({{ this._parent.get_name() }}); 

241 {% endmacro %} 

242 """) 

243 

244 def __init__(self, image, bounds, origin='upper', colormap=None, 

245 mercator_project=False, pixelated=True, 

246 name=None, overlay=True, control=True, show=True, **kwargs): 

247 super(ImageOverlay, self).__init__(name=name, overlay=overlay, 

248 control=control, show=show) 

249 self._name = 'ImageOverlay' 

250 self.bounds = bounds 

251 self.options = parse_options(**kwargs) 

252 self.pixelated = pixelated 

253 if mercator_project: 

254 image = mercator_transform( 

255 image, 

256 [bounds[0][0], bounds[1][0]], 

257 origin=origin 

258 ) 

259 

260 self.url = image_to_url(image, origin=origin, colormap=colormap) 

261 

262 def render(self, **kwargs): 

263 super(ImageOverlay, self).render() 

264 

265 figure = self.get_root() 

266 assert isinstance(figure, Figure), ('You cannot render this Element ' 

267 'if it is not in a Figure.') 

268 if self.pixelated: 

269 pixelated = """ 

270 <style> 

271 .leaflet-image-layer { 

272 /* old android/safari*/ 

273 image-rendering: -webkit-optimize-contrast; 

274 image-rendering: crisp-edges; /* safari */ 

275 image-rendering: pixelated; /* chrome */ 

276 image-rendering: -moz-crisp-edges; /* firefox */ 

277 image-rendering: -o-crisp-edges; /* opera */ 

278 -ms-interpolation-mode: nearest-neighbor; /* ie */ 

279 } 

280 </style> 

281 """ 

282 figure.header.add_child(Element(pixelated), name='leaflet-image-layer') # noqa
