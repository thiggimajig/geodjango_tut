50 """ 

51 Render the diff between this screen and the previous screen. 

52 

53 This takes two `Screen` instances. The one that represents the output like 

54 it was during the last rendering and one that represents the current 

55 output raster. Looking at these two `Screen` instances, this function will

141 def get_max_column_index(row: Dict[int, Char]) -> int: 

142 """ 

143 Return max used column index, ignoring whitespace (without style) at 

144 the end of the line. This is important for people that copy/paste 

145 terminal output. 

146 

147 There are two reasons we are sometimes seeing whitespace at the end: 

148 - `BufferControl` adds a trailing space to each line, because it's a 

149 possible cursor position, so that the line wrapping won't change if 

150 the cursor position moves around. 

151 - The `Window` adds a style class to the current line for highlighting 

152 (cursor-line). 

153 """ 

154 numbers = [ 

155 index 

156 for index, cell in row.items() 

157 if cell.char != " " or style_string_has_style[cell.style] 

158 ] 

159 numbers.append(0) 

160 return max(numbers)

162 # Render for the first time: reset styling. 

163 if not previous_screen: 

164 reset_attributes() 

165 

166 # Disable autowrap. (When entering a the alternate screen, or anytime when 

167 # we have a prompt. - In the case of a REPL, like IPython, people can have 

168 # background threads, and it's hard for debugging if their output is not 

169 # wrapped.) 

170 if not previous_screen or not full_screen: 

171 output.disable_autowrap() 

172 

173 # When the previous screen has a different size, redraw everything anyway. 

174 # Also when we are done. (We might take up less rows, so clearing is important.) 

175 if ( 

176 is_done or not previous_screen or previous_width != width 

177 ): # XXX: also consider height?? 

178 current_pos = move_cursor(Point(x=0, y=0)) 

179 reset_attributes() 

180 output.erase_down() 

181 

182 previous_screen = Screen() 

183 

184 # Get height of the screen. 

185 # (height changes as we loop over data_buffer, so remember the current value.) 

186 # (Also make sure to clip the height to the size of the output.) 

187 current_height = min(screen.height, height) 

188 

189 # Loop over the rows. 

190 row_count = min(max(screen.height, previous_screen.height), height) 

191 c = 0 # Column counter. 

192 

193 for y in range(row_count): 

194 new_row = screen.data_buffer[y] 

195 previous_row = previous_screen.data_buffer[y] 

196 zero_width_escapes_row = screen.zero_width_escapes[y] 

197 

198 new_max_line_len = min(width - 1, get_max_column_index(new_row)) 

199 previous_max_line_len = min(width - 1, get_max_column_index(previous_row)) 

200 

201 # Loop over the columns. 

202 c = 0 

203 while c <= new_max_line_len: 

204 new_char = new_row[c] 

205 old_char = previous_row[c] 

206 char_width = new_char.width or 1 

207 

208 # When the old and new character at this position are different, 

209 # draw the output. (Because of the performance, we don't call 

210 # `Char.__ne__`, but inline the same expression.) 

211 if new_char.char != old_char.char or new_char.style != old_char.style: 

212 current_pos = move_cursor(Point(x=c, y=y)) 

213 

214 # Send injected escape sequences to output. 

215 if c in zero_width_escapes_row: 

216 write_raw(zero_width_escapes_row[c]) 

217 

218 output_char(new_char) 

219 current_pos = Point(x=current_pos.x + char_width, y=current_pos.y) 

220 

221 c += char_width 

222 

223 # If the new line is shorter, trim it. 

224 if previous_screen and new_max_line_len < previous_max_line_len: 

225 current_pos = move_cursor(Point(x=new_max_line_len + 1, y=y)) 

226 reset_attributes() 

227 output.erase_end_of_line() 

228 

229 # Correctly reserve vertical space as required by the layout. 

230 # When this is a new screen (drawn for the first time), or for some reason 

231 # higher than the previous one. Move the cursor once to the bottom of the 

232 # output. That way, we're sure that the terminal scrolls up, even when the 

233 # lower lines of the canvas just contain whitespace. 

234 

235 # The most obvious reason that we actually want this behaviour is the avoid 

236 # the artifact of the input scrolling when the completion menu is shown. 

237 # (If the scrolling is actually wanted, the layout can still be build in a 

238 # way to behave that way by setting a dynamic height.) 

239 if current_height > previous_screen.height: 

240 current_pos = move_cursor(Point(x=0, y=current_height - 1)) 

241 

242 # Move cursor: 

243 if is_done: 

244 current_pos = move_cursor(Point(x=0, y=current_height)) 

245 output.erase_down() 

246 else: 

247 current_pos = move_cursor(screen.get_cursor_position(app.layout.current_window)) 

248 

249 if is_done or not full_screen: 

250 output.enable_autowrap() 

251 

252 # Always reset the color attributes. This is important because a background 

253 # thread could print data to stdout and we want that to be displayed in the 

254 # default colors. (Also, if a background color has been set, many terminals 

255 # give weird artifacts on resize events.) 

256 reset_attributes() 

257 

258 if screen.show_cursor or is_done: 

259 output.show_cursor() 

260 

261 return current_pos, last_style

459 def request_absolute_cursor_position(self) -> None: 

460 """ 

461 Get current cursor position. 

462 

463 We do this to calculate the minimum available height that we can 

464 consume for rendering the prompt. This is the available space below te 

465 cursor. 

466 

467 For vt100: Do CPR request. (answer will arrive later.) 

468 For win32: Do API call. (Answer comes immediately.) 

469 """ 

470 # Only do this request when the cursor is at the top row. (after a 

471 # clear or reset). We will rely on that in `report_absolute_cursor_row`. 

472 assert self._cursor_pos.y == 0 

473 

474 # In full-screen mode, always use the total height as min-available-height. 

475 if self.full_screen: 

476 self._min_available_height = self.output.get_size().rows 

477 return 

478 

479 # For Win32, we have an API call to get the number of rows below the 

480 # cursor. 

481 try: 

482 self._min_available_height = self.output.get_rows_below_cursor_position() 

483 return 

484 except NotImplementedError: 

485 pass 

486 

487 # Use CPR. 

488 if self.cpr_support == CPR_Support.NOT_SUPPORTED: 

489 return

491 def do_cpr() -> None: 

492 # Asks for a cursor position report (CPR). 

493 self._waiting_for_cpr_futures.append(Future()) 

494 self.output.ask_for_cpr() 

495 

496 if self.cpr_support == CPR_Support.SUPPORTED: 

497 do_cpr() 

498 return 

499 

500 # If we don't know whether CPR is supported, only do a request if 

501 # none is pending, and test it, using a timer. 

502 if self.waiting_for_cpr: 

503 return 

504 

505 do_cpr() 

506 

507 async def timer() -> None: 

508 await sleep(self.CPR_TIMEOUT) 

509 

510 # Not set in the meantime -> not supported. 

511 if self.cpr_support == CPR_Support.UNKNOWN: 

512 self.cpr_support = CPR_Support.NOT_SUPPORTED 

513 

514 if self.cpr_not_supported_callback: 

515 # Make sure to call this callback in the main thread. 

516 self.cpr_not_supported_callback() 

517 

518 get_app().create_background_task(timer())

520 def report_absolute_cursor_row(self, row: int) -> None: 

521 """ 

522 To be called when we know the absolute cursor position. 

523 (As an answer of a "Cursor Position Request" response.) 

524 """ 

525 self.cpr_support = CPR_Support.SUPPORTED 

526 

527 # Calculate the amount of rows from the cursor position until the 

528 # bottom of the terminal. 

529 total_rows = self.output.get_size().rows 

530 rows_below_cursor = total_rows - row + 1 

531 

532 # Set the minimum available height. 

533 self._min_available_height = rows_below_cursor 

534 

535 # Pop and set waiting for CPR future. 

536 try: 

537 f = self._waiting_for_cpr_futures.popleft() 

538 except IndexError: 

539 pass # Received CPR response without having a CPR. 

540 else: 

541 f.set_result(None)

543 @property 

544 def waiting_for_cpr(self) -> bool: 

545 """ 

546 Waiting for CPR flag. True when we send the request, but didn't got a 

547 response. 

548 """ 

549 return bool(self._waiting_for_cpr_futures)

551 async def wait_for_cpr_responses(self, timeout: int = 1) -> None: 

552 """ 

553 Wait for a CPR response. 

554 """ 

555 cpr_futures = list(self._waiting_for_cpr_futures) # Make copy. 

556 

557 # When there are no CPRs in the queue. Don't do anything. 

558 if not cpr_futures or self.cpr_support == CPR_Support.NOT_SUPPORTED: 

559 return None 

560 

561 async def wait_for_responses() -> None: 

562 for response_f in cpr_futures: 

563 await response_f 

564 

565 async def wait_for_timeout() -> None: 

566 await sleep(timeout) 

567 

568 # Got timeout, erase queue. 

569 for response_f in cpr_futures: 

570 response_f.cancel() 

571 self._waiting_for_cpr_futures = deque() 

572 

573 tasks = { 

574 ensure_future(wait_for_responses()), 

575 ensure_future(wait_for_timeout()), 

576 } 

577 _, pending = await wait(tasks, return_when=FIRST_COMPLETED) 

578 for task in pending: 

579 task.cancel()

581 def render( 

582 self, app: "Application[Any]", layout: "Layout", is_done: bool = False 

583 ) -> None: 

584 """ 

585 Render the current interface to the output. 

586 

587 :param is_done: When True, put the cursor at the end of the interface. We 

588 won't print any changes to this part. 

589 """ 

590 output = self.output 

591 

592 # Enter alternate screen. 

593 if self.full_screen and not self._in_alternate_screen: 

594 self._in_alternate_screen = True 

595 output.enter_alternate_screen() 

596 

597 # Enable bracketed paste. 

598 if not self._bracketed_paste_enabled: 

599 self.output.enable_bracketed_paste() 

600 self._bracketed_paste_enabled = True 

601 

602 # Reset cursor key mode. 

603 if not self._cursor_key_mode_reset: 

604 self.output.reset_cursor_key_mode() 

605 self._cursor_key_mode_reset = True 

606 

607 # Enable/disable mouse support. 

608 needs_mouse_support = self.mouse_support() 

609 

610 if needs_mouse_support and not self._mouse_support_enabled: 

611 output.enable_mouse_support() 

612 self._mouse_support_enabled = True 

613 

614 elif not needs_mouse_support and self._mouse_support_enabled: 

615 output.disable_mouse_support() 

616 self._mouse_support_enabled = False 

617 

618 # Create screen and write layout to it. 

619 size = output.get_size() 

620 screen = Screen() 

621 screen.show_cursor = False # Hide cursor by default, unless one of the 

622 # containers decides to display it. 

623 mouse_handlers = MouseHandlers() 

624 

625 # Calculate height. 

626 if self.full_screen: 

627 height = size.rows 

628 elif is_done: 

629 # When we are done, we don't necessary want to fill up until the bottom. 

630 height = layout.container.preferred_height( 

631 size.columns, size.rows 

632 ).preferred 

633 else: 

634 last_height = self._last_screen.height if self._last_screen else 0 

635 height = max( 

636 self._min_available_height, 

637 last_height, 

638 layout.container.preferred_height(size.columns, size.rows).preferred, 

639 ) 

640 

641 height = min(height, size.rows) 

642 

643 # When the size changes, don't consider the previous screen. 

644 if self._last_size != size: 

645 self._last_screen = None 

646 

647 # When we render using another style or another color depth, do a full 

648 # repaint. (Forget about the previous rendered screen.) 

649 # (But note that we still use _last_screen to calculate the height.) 

650 if ( 

651 self.style.invalidation_hash() != self._last_style_hash 

652 or app.style_transformation.invalidation_hash() 

653 != self._last_transformation_hash 

654 or app.color_depth != self._last_color_depth 

655 ): 

656 self._last_screen = None 

657 self._attrs_for_style = None 

658 self._style_string_has_style = None

752def print_formatted_text( 

753 output: Output, 

754 formatted_text: AnyFormattedText, 

755 style: BaseStyle, 

756 style_transformation: Optional[StyleTransformation] = None, 

757 color_depth: Optional[ColorDepth] = None, 

758) -> None: 

759 """ 

760 Print a list of (style_str, text) tuples in the given style to the output. 

761 """ 

762 fragments = to_formatted_text(formatted_text) 

763 style_transformation = style_transformation or DummyStyleTransformation() 

764 color_depth = color_depth or output.get_default_color_depth() 

765 

766 # Reset first. 

767 output.reset_attributes() 

768 output.enable_autowrap() 

769 last_attrs: Optional[Attrs] = None 

770 

771 # Print all (style_str, text) tuples. 

772 attrs_for_style_string = _StyleStringToAttrsCache( 

773 style.get_attrs_for_style_str, style_transformation 

774 ) 

775 

776 for style_str, text, *_ in fragments: 

777 attrs = attrs_for_style_string[style_str] 

778 

779 # Set style attributes if something changed. 

780 if attrs != last_attrs: 

781 if attrs: 

782 output.set_attributes(attrs, color_depth) 

783 else: 

784 output.reset_attributes() 

785 last_attrs = attrs 

786 

787 # Eliminate carriage returns 

788 text = text.replace("\r", "") 

789 

790 # Assume that the output is raw, and insert a carriage return before 

791 # every newline. (Also important when the front-end is a telnet client.) 

792 output.write(text.replace("\n", "\r\n")) 

793 

794 # Reset again. 

795 output.reset_attributes() 

796 output.flush()
