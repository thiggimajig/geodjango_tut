28class PostGISOperator(SpatialOperator): 

29 def __init__(self, geography=False, raster=False, **kwargs): 

30 # Only a subset of the operators and functions are available for the 

31 # geography type. 

32 self.geography = geography 

33 # Only a subset of the operators and functions are available for the 

34 # raster type. Lookups that don't support raster will be converted to 

35 # polygons. If the raster argument is set to BILATERAL, then the 

36 # operator cannot handle mixed geom-raster lookups. 

37 self.raster = raster 

38 super().__init__(**kwargs)

40 def as_sql(self, connection, lookup, template_params, *args): 

41 if lookup.lhs.output_field.geography and not self.geography: 

42 raise ValueError('PostGIS geography does not support the "%s" ' 

43 'function/operator.' % (self.func or self.op,)) 

44 

45 template_params = self.check_raster(lookup, template_params) 

46 return super().as_sql(connection, lookup, template_params, *args)

48 def check_raster(self, lookup, template_params): 

49 spheroid = lookup.rhs_params and lookup.rhs_params[-1] == 'spheroid' 

50 

51 # Check which input is a raster. 

52 lhs_is_raster = lookup.lhs.field.geom_type == 'RASTER' 

53 rhs_is_raster = isinstance(lookup.rhs, GDALRaster) 

54 

55 # Look for band indices and inject them if provided. 

56 if lookup.band_lhs is not None and lhs_is_raster: 

57 if not self.func: 

58 raise ValueError('Band indices are not allowed for this operator, it works on bbox only.') 

59 template_params['lhs'] = '%s, %s' % (template_params['lhs'], lookup.band_lhs) 

60 

61 if lookup.band_rhs is not None and rhs_is_raster: 

62 if not self.func: 

63 raise ValueError('Band indices are not allowed for this operator, it works on bbox only.') 

64 template_params['rhs'] = '%s, %s' % (template_params['rhs'], lookup.band_rhs) 

65 

66 # Convert rasters to polygons if necessary. 

67 if not self.raster or spheroid: 

68 # Operators without raster support. 

69 if lhs_is_raster: 

70 template_params['lhs'] = 'ST_Polygon(%s)' % template_params['lhs'] 

71 if rhs_is_raster: 

72 template_params['rhs'] = 'ST_Polygon(%s)' % template_params['rhs'] 

73 elif self.raster == BILATERAL: 

74 # Operators with raster support but don't support mixed (rast-geom) 

75 # lookups. 

76 if lhs_is_raster and not rhs_is_raster: 

77 template_params['lhs'] = 'ST_Polygon(%s)' % template_params['lhs'] 

78 elif rhs_is_raster and not lhs_is_raster: 

79 template_params['rhs'] = 'ST_Polygon(%s)' % template_params['rhs'] 

80 

81 return template_params

160 def spatial_version(self): 

161 """Determine the version of the PostGIS library.""" 

162 # Trying to get the PostGIS version because the function 

163 # signatures will depend on the version used. The cost 

164 # here is a database query to determine the version, which 

165 # can be mitigated by setting `POSTGIS_VERSION` with a 3-tuple 

166 # comprising user-supplied values for the major, minor, and 

167 # subminor revision of PostGIS. 

168 if hasattr(settings, 'POSTGIS_VERSION'): 

169 version = settings.POSTGIS_VERSION 

170 else: 

171 # Run a basic query to check the status of the connection so we're 

172 # sure we only raise the error below if the problem comes from 

173 # PostGIS and not from PostgreSQL itself (see #24862). 

174 self._get_postgis_func('version') 

175 

176 try: 

177 vtup = self.postgis_version_tuple() 

178 except ProgrammingError: 

179 raise ImproperlyConfigured( 

180 'Cannot determine PostGIS version for database "%s" ' 

181 'using command "SELECT postgis_lib_version()". ' 

182 'GeoDjango requires at least PostGIS version 2.3. ' 

183 'Was the database created from a spatial database ' 

184 'template?' % self.connection.settings_dict['NAME'] 

185 ) 

186 version = vtup[1:] 

187 return version

189 def convert_extent(self, box): 

190 """ 

191 Return a 4-tuple extent for the `Extent` aggregate by converting 

192 the bounding box text returned by PostGIS (`box` argument), for 

193 example: "BOX(-90.0 30.0, -85.0 40.0)". 

194 """ 

195 if box is None: 

196 return None 

197 ll, ur = box[4:-1].split(',') 

198 xmin, ymin = map(float, ll.split()) 

199 xmax, ymax = map(float, ur.split()) 

200 return (xmin, ymin, xmax, ymax)

202 def convert_extent3d(self, box3d): 

203 """ 

204 Return a 6-tuple extent for the `Extent3D` aggregate by converting 

205 the 3d bounding-box text returned by PostGIS (`box3d` argument), for 

206 example: "BOX3D(-90.0 30.0 1, -85.0 40.0 2)". 

207 """ 

208 if box3d is None: 

209 return None 

210 ll, ur = box3d[6:-1].split(',') 

211 xmin, ymin, zmin = map(float, ll.split()) 

212 xmax, ymax, zmax = map(float, ur.split()) 

213 return (xmin, ymin, zmin, xmax, ymax, zmax)

215 def geo_db_type(self, f): 

216 """ 

217 Return the database field type for the given spatial field. 

218 """ 

219 if f.geom_type == 'RASTER': 

220 return 'raster'

236 def get_distance(self, f, dist_val, lookup_type): 

237 """ 

238 Retrieve the distance parameters for the given geometry field, 

239 distance lookup value, and the distance lookup type. 

240 

241 This is the most complex implementation of the spatial backends due to 

242 what is supported on geodetic geometry columns vs. what's available on 

243 projected geometry columns. In addition, it has to take into account 

244 the geography column type. 

245 """ 

246 # Getting the distance parameter 

247 value = dist_val[0] 

248 

249 # Shorthand boolean flags. 

250 geodetic = f.geodetic(self.connection) 

251 geography = f.geography 

252 

253 if isinstance(value, Distance): 

254 if geography: 

255 dist_param = value.m 

256 elif geodetic: 

257 if lookup_type == 'dwithin': 

258 raise ValueError('Only numeric values of degree units are ' 

259 'allowed on geographic DWithin queries.') 

260 dist_param = value.m 

261 else: 

262 dist_param = getattr(value, Distance.unit_attname(f.units_name(self.connection))) 

263 else: 

264 # Assuming the distance is in the units of the field. 

265 dist_param = value 

266 

267 return [dist_param]

335 def proj_version_tuple(self): 

336 """ 

337 Return the version of PROJ used by PostGIS as a tuple of the 

338 major, minor, and subminor release numbers. 

339 """ 

340 proj_regex = re.compile(r'(\d+)\.(\d+)\.(\d+)') 

341 proj_ver_str = self.postgis_proj_version() 

342 m = proj_regex.search(proj_ver_str) 

343 if m: 

344 return tuple(map(int, m.groups()))
