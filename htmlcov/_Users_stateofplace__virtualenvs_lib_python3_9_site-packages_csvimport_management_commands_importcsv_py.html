464 def type_clean(self, field, value, loglist, row=0): 

465 """ Data value clean up - type formatting""" 

466 if not self.fieldmap.get(field): 

467 raise Exception( 

468 "Fieldmap is not populated for %s -\n%s" % (field, self.fieldmap) 

469 ) 

470 field_type = self.fieldmap.get(field).get_internal_type() 

471 

472 try: 

473 value = value.strip() 

474 except AttributeError: 

475 pass 

476 

477 # Tidy up boolean data 

478 if field_type in BOOLEAN: 

479 value = value in BOOLEAN_TRUE 

480 # sqlite fix since it just uses int under the hood 

481 if self.db_backend in SMALLINT_DBS: 

482 if value: 

483 value = 1 

484 else: 

485 value = 0 

486 

487 # Tidy up numeric data 

488 if field_type in NUMERIC: 

489 if not value: 

490 value = 0 

491 else: 

492 try: 

493 value = float(value) 

494 except: 

495 loglist.append( 

496 "row %s: Column %s = %s is not a number so is set to 0" 

497 % (row, field, value) 

498 ) 

499 value = 0 

500 if field_type in INTEGER: 

501 # 1e+28 = 9999999999999999583119736832L 

502 if value > 9223372036854775807: 

503 intmsg = ( 

504 "row %s: Column %s = %s more than the max integer 9223372036854775807" 

505 % (row, field, value) 

506 ) 

507 if self.db_backend in SMALLINT_DBS: 

508 intmsg += " sqlite may error with big integers so rounded down" 

509 value = 9223372036854775807 

510 loglist.append(intmsg) 

511 if str(value).lower() in ("nan", "inf", "+inf", "-inf"): 

512 loglist.append( 

513 "row %s: Column %s = %s is not an integer so is set to 0" 

514 % (row, field, value) 

515 ) 

516 value = 0 

517 value = int(value) 

518 if value < 0 and field_type.startswith("Positive"): 

519 loglist.append( 

520 "row %s: Column %s = %s, less than zero so set to 0" 

521 % (row, field, value) 

522 ) 

523 value = 0 

524 # date data - remove the date if it doesn't convert so null=True can work 

525 if field_type in DATE:

539 # loglist.append('row %s: Column %s = %s not date format' % (i, field, value)) 

540 value = None 

541 return value

543 def parse_header(self, headlist): 

544 """ Parse the list of headings and match with self.fieldmap """ 

545 mapping = [] 

546 found = [] 

547 headlist = [cleancol.sub("_", col) for col in headlist] 

548 logstr = "" 

549 for i, heading in enumerate(headlist): 

550 for key in ( 

551 (heading, heading.lower()) if heading != heading.lower() else (heading,) 

552 ): 

553 if key in self.fieldmap: 

554 found.append(key.lower()) 

555 field = self.fieldmap[key] 

556 key = self.check_fkey(key, field) 

557 mapping.append("column%s=%s" % (i + 1, key)) 

558 for key in headlist: 

559 if key.lower() not in found: 

560 logstr += ", %s" % key 

561 if mapping: 

562 mappingstr = ",".join(mapping) 

563 if logstr: 

564 logmsg = "CSV header unmatched ignored cols = %s" % logstr[2:] 

565 else: 

566 logmsg = "CSV header matched all cols" 

567 self.loglist.append(logmsg) 

568 return mappingstr 

569 return ""

571 def insert_fkey(self, foreignkey, rowcol): 

572 """ Add fkey if not present 

573 If there is corresponding data in the model already, 

574 we do not need to add more, since we are dealing with 

575 foreign keys, therefore foreign data 

576 """ 

577 fk_key, fk_field = foreignkey 

578 if fk_key and fk_field: 

579 # Allow users to specify app label for fk model if they want 

580 if fk_key.find(".") > -1: 

581 new_app_label, fk_key = fk_key.split(".") 

582 else: 

583 try: 

584 new_app_label = ContentType.objects.get(model=fk_key).app_label 

585 except: 

586 new_app_label = self.app_label 

587 fk_model = get_model(new_app_label, fk_key) 

588 matches = fk_model.objects.filter(**{fk_field + "__exact": rowcol}) 

589 

590 if not matches: 

591 key = fk_model() 

592 key.__setattr__(fk_field, rowcol) 

593 key.save() 

594 

595 rowcol = fk_model.objects.filter(**{fk_field + "__exact": rowcol})[0] 

596 return rowcol
