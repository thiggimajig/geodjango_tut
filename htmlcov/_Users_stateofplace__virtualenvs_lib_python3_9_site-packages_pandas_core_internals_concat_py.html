105def concat_arrays(to_concat: list) -> ArrayLike: 

106 """ 

107 Alternative for concat_compat but specialized for use in the ArrayManager. 

108 

109 Differences: only deals with 1D arrays (no axis keyword), assumes 

110 ensure_wrapped_if_datetimelike and does not skip empty arrays to determine 

111 the dtype. 

112 In addition ensures that all NullArrayProxies get replaced with actual 

113 arrays. 

114 

115 Parameters 

116 ---------- 

117 to_concat : list of arrays 

118 

119 Returns 

120 ------- 

121 np.ndarray or ExtensionArray 

122 """ 

123 # ignore the all-NA proxies to determine the resulting dtype 

124 to_concat_no_proxy = [x for x in to_concat if not isinstance(x, NullArrayProxy)] 

125 

126 dtypes = {x.dtype for x in to_concat_no_proxy} 

127 single_dtype = len(dtypes) == 1 

128 

129 if single_dtype: 

130 target_dtype = to_concat_no_proxy[0].dtype 

131 elif all(x.kind in ["i", "u", "b"] and isinstance(x, np.dtype) for x in dtypes): 

132 # GH#42092 

133 target_dtype = np.find_common_type(list(dtypes), []) 

134 else: 

135 target_dtype = find_common_type([arr.dtype for arr in to_concat_no_proxy]) 

136 

137 if target_dtype.kind in ["m", "M"]: 

138 # for datetimelike use DatetimeArray/TimedeltaArray concatenation 

139 # don't use arr.astype(target_dtype, copy=False), because that doesn't 

140 # work for DatetimeArray/TimedeltaArray (returns ndarray) 

141 to_concat = [ 

142 arr.to_array(target_dtype) if isinstance(arr, NullArrayProxy) else arr 

143 for arr in to_concat 

144 ] 

145 return type(to_concat_no_proxy[0])._concat_same_type(to_concat, axis=0) 

146 

147 to_concat = [ 

148 arr.to_array(target_dtype) 

149 if isinstance(arr, NullArrayProxy) 

150 else cast_to_common_type(arr, target_dtype) 

151 for arr in to_concat 

152 ] 

153 

154 if isinstance(to_concat[0], ExtensionArray): 

155 cls = type(to_concat[0]) 

156 return cls._concat_same_type(to_concat) 

157 

158 result = np.concatenate(to_concat) 

159 

160 # TODO decide on exact behaviour (we shouldn't do this only for empty result) 

161 # see https://github.com/pandas-dev/pandas/issues/39817 

162 if len(result) == 0: 

163 # all empties -> check for bool to not coerce to float 

164 kinds = {obj.dtype.kind for obj in to_concat_no_proxy} 

165 if len(kinds) != 1: 

166 if "b" in kinds: 

167 result = result.astype(object) 

168 return result

239def _get_mgr_concatenation_plan(mgr: BlockManager, indexers: dict[int, np.ndarray]): 

240 """ 

241 Construct concatenation plan for given block manager and indexers. 

242 

243 Parameters 

244 ---------- 

245 mgr : BlockManager 

246 indexers : dict of {axis: indexer} 

247 

248 Returns 

249 ------- 

250 plan : list of (BlockPlacement, JoinUnit) tuples 

251 

252 """ 

253 # Calculate post-reindex shape , save for item axis which will be separate 

254 # for each block anyway. 

255 mgr_shape_list = list(mgr.shape) 

256 for ax, indexer in indexers.items(): 

257 mgr_shape_list[ax] = len(indexer) 

258 mgr_shape = tuple(mgr_shape_list) 

259 

260 if 0 in indexers: 

261 ax0_indexer = indexers.pop(0) 

262 blknos = algos.take_nd(mgr.blknos, ax0_indexer, fill_value=-1) 

263 blklocs = algos.take_nd(mgr.blklocs, ax0_indexer, fill_value=-1) 

264 else: 

265 

266 if mgr.is_single_block: 

267 blk = mgr.blocks[0] 

268 return [(blk.mgr_locs, JoinUnit(blk, mgr_shape, indexers))] 

269 

270 # error: Incompatible types in assignment (expression has type "None", variable 

271 # has type "ndarray") 

272 ax0_indexer = None # type: ignore[assignment] 

273 blknos = mgr.blknos 

274 blklocs = mgr.blklocs 

275 

276 plan = [] 

277 for blkno, placements in libinternals.get_blkno_placements(blknos, group=False): 

278 

279 assert placements.is_slice_like 

280 

281 join_unit_indexers = indexers.copy() 

282 

283 shape_list = list(mgr_shape) 

284 shape_list[0] = len(placements) 

285 shape = tuple(shape_list) 

286 

287 if blkno == -1: 

288 unit = JoinUnit(None, shape) 

289 else: 

290 blk = mgr.blocks[blkno] 

291 ax0_blk_indexer = blklocs[placements.indexer] 

292 

293 unit_no_ax0_reindexing = ( 

294 len(placements) == len(blk.mgr_locs) 

295 and 

296 # Fastpath detection of join unit not 

297 # needing to reindex its block: no ax0 

298 # reindexing took place and block 

299 # placement was sequential before. 

300 ( 

301 ( 

302 ax0_indexer is None 

303 and blk.mgr_locs.is_slice_like 

304 and blk.mgr_locs.as_slice.step == 1 

305 ) 

306 or 

307 # Slow-ish detection: all indexer locs 

308 # are sequential (and length match is 

309 # checked above). 

310 (np.diff(ax0_blk_indexer) == 1).all() 

311 ) 

312 ) 

313 

314 # Omit indexer if no item reindexing is required. 

315 if unit_no_ax0_reindexing: 

316 join_unit_indexers.pop(0, None) 

317 else: 

318 join_unit_indexers[0] = ax0_blk_indexer 

319 

320 unit = JoinUnit(blk, shape, join_unit_indexers) 

321 

322 plan.append((placements, unit)) 

323 

324 return plan

475def _concatenate_join_units( 

476 join_units: list[JoinUnit], concat_axis: int, copy: bool 

477) -> ArrayLike: 

478 """ 

479 Concatenate values from several join units along selected axis. 

480 """ 

481 if concat_axis == 0 and len(join_units) > 1: 

482 # Concatenating join units along ax0 is handled in _merge_blocks. 

483 raise AssertionError("Concatenating join units along axis0") 

484 

485 empty_dtype = _get_empty_dtype(join_units) 

486 

487 has_none_blocks = any(unit.block is None for unit in join_units) 

488 upcasted_na = _dtype_to_na_value(empty_dtype, has_none_blocks) 

489 

490 to_concat = [ 

491 ju.get_reindexed_values(empty_dtype=empty_dtype, upcasted_na=upcasted_na) 

492 for ju in join_units 

493 ] 

494 

495 if len(to_concat) == 1: 

496 # Only one block, nothing to concatenate. 

497 concat_values = to_concat[0] 

498 if copy: 

499 if isinstance(concat_values, np.ndarray): 

500 # non-reindexed (=not yet copied) arrays are made into a view 

501 # in JoinUnit.get_reindexed_values 

502 if concat_values.base is not None: 

503 concat_values = concat_values.copy() 

504 else: 

505 concat_values = concat_values.copy() 

506 

507 elif any(is_1d_only_ea_obj(t) for t in to_concat): 

508 # TODO(EA2D): special case not needed if all EAs used HybridBlocks 

509 # NB: we are still assuming here that Hybrid blocks have shape (1, N) 

510 # concatting with at least one EA means we are concatting a single column 

511 # the non-EA values are 2D arrays with shape (1, n) 

512 

513 # error: Invalid index type "Tuple[int, slice]" for 

514 # "Union[ExtensionArray, ndarray]"; expected type "Union[int, slice, ndarray]" 

515 to_concat = [ 

516 t if is_1d_only_ea_obj(t) else t[0, :] # type: ignore[index] 

517 for t in to_concat 

518 ] 

519 concat_values = concat_compat(to_concat, axis=0, ea_compat_axis=True) 

520 concat_values = ensure_block_shape(concat_values, 2) 

521 

522 else: 

523 concat_values = concat_compat(to_concat, axis=concat_axis) 

524 

525 return concat_values

528def _dtype_to_na_value(dtype: DtypeObj, has_none_blocks: bool): 

529 """ 

530 Find the NA value to go with this dtype. 

531 """ 

532 if isinstance(dtype, ExtensionDtype): 

533 return dtype.na_value 

534 elif dtype.kind in ["m", "M"]: 

535 return dtype.type("NaT") 

536 elif dtype.kind in ["f", "c"]: 

537 return dtype.type("NaN") 

538 elif dtype.kind == "b": 

539 # different from missing.na_value_for_dtype 

540 return None 

541 elif dtype.kind in ["i", "u"]: 

542 if not has_none_blocks: 

543 # different from missing.na_value_for_dtype 

544 return None 

545 return np.nan 

546 elif dtype.kind == "O": 

547 return np.nan 

548 raise NotImplementedError

626 """

628

652def _combine_concat_plans(plans, concat_axis: int): 

653 """ 

654 Combine multiple concatenation plans into one. 

655 

656 existing_plan is updated in-place. 

657 """ 

658 if len(plans) == 1: 

659 for p in plans[0]: 

660 yield p[0], [p[1]] 

661 

662 elif concat_axis == 0: 

663 offset = 0 

664 for plan in plans: 

665 last_plc = None 

666 

667 for plc, unit in plan: 

668 yield plc.add(offset), [unit] 

669 last_plc = plc 

670 

671 if last_plc is not None: 

672 offset += last_plc.as_slice.stop 

673 

674 else: 

675 num_ended = [0] 

676 

677 def _next_or_none(seq): 

678 retval = next(seq, None) 

679 if retval is None: 

680 num_ended[0] += 1 

681 return retval 

682 

683 plans = list(map(iter, plans)) 

684 next_items = list(map(_next_or_none, plans)) 

685 

686 while num_ended[0] != len(next_items): 

687 if num_ended[0] > 0: 

688 raise ValueError("Plan shapes are not aligned") 

689 

690 placements, units = zip(*next_items) 

691 

692 lengths = list(map(len, placements)) 

693 min_len, max_len = min(lengths), max(lengths) 

694 

695 if min_len == max_len: 

696 yield placements[0], units 

697 next_items[:] = map(_next_or_none, plans) 

698 else: 

699 yielded_placement = None 

700 yielded_units = [None] * len(next_items) 

701 for i, (plc, unit) in enumerate(next_items): 

702 yielded_units[i] = unit 

703 if len(plc) > min_len: 

704 # _trim_join_unit updates unit in place, so only 

705 # placement needs to be sliced to skip min_len. 

706 next_items[i] = (plc[min_len:], _trim_join_unit(unit, min_len)) 

707 else: 

708 yielded_placement = plc 

709 next_items[i] = _next_or_none(plans[i]) 

710 

711 yield yielded_placement, yielded_units
