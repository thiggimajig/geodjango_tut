10def filefind(filename, path_dirs=None): 

11 """Find a file by looking through a sequence of paths. 

12 

13 This iterates through a sequence of paths looking for a file and returns 

14 the full, absolute path of the first occurence of the file. If no set of 

15 path dirs is given, the filename is tested as is, after running through 

16 :func:`expandvars` and :func:`expanduser`. Thus a simple call:: 

17 

18 filefind('myfile.txt') 

19 

20 will find the file in the current working dir, but:: 

21 

22 filefind('~/myfile.txt') 

23 

24 Will find the file in the users home directory. This function does not 

25 automatically try any paths, such as the cwd or the user's home directory. 

26 

27 Parameters 

28 ---------- 

29 filename : str 

30 The filename to look for. 

31 path_dirs : str, None or sequence of str 

32 The sequence of paths to look for the file in. If None, the filename 

33 need to be absolute or be in the cwd. If a string, the string is 

34 put into a sequence and the searched. If a sequence, walk through 

35 each element and join with ``filename``, calling :func:`expandvars` 

36 and :func:`expanduser` before testing for existence. 

37 

38 Returns 

39 ------- 

40 Raises :exc:`IOError` or returns absolute path to file. 

41 """ 

42 

43 # If paths are quoted, abspath gets confused, strip them... 

44 filename = filename.strip('"').strip("'") 

45 # If the input is an absolute path, just check it exists 

46 if os.path.isabs(filename) and os.path.isfile(filename): 

47 return filename 

48 

49 if path_dirs is None: 

50 path_dirs = ("",) 

51 elif isinstance(path_dirs, str): 

52 path_dirs = (path_dirs,) 

53 

54 for path in path_dirs: 

55 if path == ".": 

56 path = os.getcwd() 

57 testname = expand_path(os.path.join(path, filename)) 

58 if os.path.isfile(testname): 

59 return os.path.abspath(testname) 

60 

61 raise IOError( 

62 "File %r does not exist in any of the search paths: %r" % (filename, path_dirs) 

63 )

66def expand_path(s): 

67 """Expand $VARS and ~names in a string, like a shell 

68 

69 :Examples: 

70 

71 In [2]: os.environ['FOO']='test' 

72 

73 In [3]: expand_path('variable FOO is $FOO') 

74 Out[3]: 'variable FOO is test' 

75 """ 

76 # This is a pretty subtle hack. When expand user is given a UNC path 

77 # on Windows (\\server\share$\%username%), os.path.expandvars, removes 

78 # the $ to get (\\server\share\%username%). I think it considered $ 

79 # alone an empty var. But, we need the $ to remains there (it indicates 

80 # a hidden share). 

81 if os.name == "nt": 

82 s = s.replace("$\\", "IPYTHON_TEMP") 

83 s = os.path.expandvars(os.path.expanduser(s)) 

84 if os.name == "nt": 

85 s = s.replace("IPYTHON_TEMP", "$\\") 

86 return s
