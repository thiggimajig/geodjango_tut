26class ASTTokens(object): 

27 """ 

28 ASTTokens maintains the text of Python code in several forms: as a string, as line numbers, and 

29 as tokens, and is used to mark and access token and position information. 

30 

31 ``source_text`` must be a unicode or UTF8-encoded string. If you pass in UTF8 bytes, remember 

32 that all offsets you'll get are to the unicode text, which is available as the ``.text`` 

33 property. 

34 

35 If ``parse`` is set, the ``source_text`` will be parsed with ``ast.parse()``, and the resulting 

36 tree marked with token info and made available as the ``.tree`` property. 

37 

38 If ``tree`` is given, it will be marked and made available as the ``.tree`` property. In 

39 addition to the trees produced by the ``ast`` module, ASTTokens will also mark trees produced 

40 using ``astroid`` library <https://www.astroid.org>. 

41 

42 If only ``source_text`` is given, you may use ``.mark_tokens(tree)`` to mark the nodes of an AST 

43 tree created separately. 

44 """ 

45 def __init__(self, source_text, parse=False, tree=None, filename='<unknown>'): 

46 self._filename = filename 

47 self._tree = ast.parse(source_text, filename) if parse else tree 

48 

49 # Decode source after parsing to let Python 2 handle coding declarations. 

50 # (If the encoding was not utf-8 compatible, then even if it parses correctly, 

51 # we'll fail with a unicode error here.) 

52 if isinstance(source_text, six.binary_type): 

53 source_text = source_text.decode('utf8') 

54 

55 self._text = source_text 

56 self._line_numbers = LineNumbers(source_text) 

57 

58 # Tokenize the code. 

59 self._tokens = list(self._generate_tokens(source_text)) 

60 

61 # Extract the start positions of all tokens, so that we can quickly map positions to tokens. 

62 self._token_offsets = [tok.startpos for tok in self._tokens] 

63 

64 if self._tree: 

65 self.mark_tokens(self._tree) 

66 

67 

68 def mark_tokens(self, root_node): 

69 """ 

70 Given the root of the AST or Astroid tree produced from source_text, visits all nodes marking 

71 them with token and position information by adding ``.first_token`` and 

72 ``.last_token``attributes. This is done automatically in the constructor when ``parse`` or 

73 ``tree`` arguments are set, but may be used manually with a separate AST or Astroid tree. 

74 """ 

75 # The hard work of this class is done by MarkTokens 

76 MarkTokens(self).visit_tree(root_node)

79 def _generate_tokens(self, text): 

80 """ 

81 Generates tokens for the given code. 

82 """ 

83 # This is technically an undocumented API for Python3, but allows us to use the same API as for 

84 # Python2. See http://stackoverflow.com/a/4952291/328565. 

85 for index, tok in enumerate(tokenize.generate_tokens(io.StringIO(text).readline)): 

86 tok_type, tok_str, start, end, line = tok 

87 yield Token(tok_type, tok_str, start, end, line, index, 

88 self._line_numbers.line_to_offset(start[0], start[1]), 

89 self._line_numbers.line_to_offset(end[0], end[1])) 

90

94 return self._text

104 return self._tree

109 return self._filename

111 def get_token_from_offset(self, offset): 

112 """ 

113 Returns the token containing the given character offset (0-based position in source text), 

114 or the preceeding token if the position is between tokens. 

115 """ 

116 return self._tokens[bisect.bisect(self._token_offsets, offset) - 1] 

117 

118 def get_token(self, lineno, col_offset): 

119 """ 

120 Returns the token containing the given (lineno, col_offset) position, or the preceeding token 

121 if the position is between tokens. 

122 """ 

123 # TODO: add test for multibyte unicode. We need to translate offsets from ast module (which 

124 # are in utf8) to offsets into the unicode text. tokenize module seems to use unicode offsets 

125 # but isn't explicit. 

126 return self.get_token_from_offset(self._line_numbers.line_to_offset(lineno, col_offset)) 

127 

128 def get_token_from_utf8(self, lineno, col_offset): 

129 """ 

130 Same as get_token(), but interprets col_offset as a UTF8 offset, which is what `ast` uses. 

131 """ 

132 return self.get_token(lineno, self._line_numbers.from_utf8_col(lineno, col_offset))

134 def next_token(self, tok, include_extra=False): 

135 """ 

136 Returns the next token after the given one. If include_extra is True, includes non-coding 

137 tokens from the tokenize module, such as NL and COMMENT. 

138 """ 

139 i = tok.index + 1 

140 if not include_extra: 

141 while is_non_coding_token(self._tokens[i].type): 

142 i += 1 

143 return self._tokens[i] 

144 

145 def prev_token(self, tok, include_extra=False): 

146 """ 

147 Returns the previous token before the given one. If include_extra is True, includes non-coding 

148 tokens from the tokenize module, such as NL and COMMENT. 

149 """ 

150 i = tok.index - 1 

151 if not include_extra: 

152 while is_non_coding_token(self._tokens[i].type): 

153 i -= 1 

154 return self._tokens[i]

156 def find_token(self, start_token, tok_type, tok_str=None, reverse=False): 

157 """ 

158 Looks for the first token, starting at start_token, that matches tok_type and, if given, the 

159 token string. Searches backwards if reverse is True. Returns ENDMARKER token if not found (you 

160 can check it with `token.ISEOF(t.type)`. 

161 """ 

162 t = start_token 

163 advance = self.prev_token if reverse else self.next_token 

164 while not match_token(t, tok_type, tok_str) and not token.ISEOF(t.type): 

165 t = advance(t, include_extra=True) 

166 return t

168 def token_range(self, first_token, last_token, include_extra=False): 

169 """ 

170 Yields all tokens in order from first_token through and including last_token. If 

171 include_extra is True, includes non-coding tokens such as tokenize.NL and .COMMENT. 

172 """ 

173 for i in xrange(first_token.index, last_token.index + 1): 

174 if include_extra or not is_non_coding_token(self._tokens[i].type): 

175 yield self._tokens[i] 

176 

177 def get_tokens(self, node, include_extra=False): 

178 """ 

179 Yields all tokens making up the given node. If include_extra is True, includes non-coding 

180 tokens such as tokenize.NL and .COMMENT. 

181 """ 

182 return self.token_range(node.first_token, node.last_token, include_extra=include_extra) 

183 

184 def get_text_range(self, node): 

185 """ 

186 After mark_tokens() has been called, returns the (startpos, endpos) positions in source text 

187 corresponding to the given node. Returns (0, 0) for nodes (like `Load`) that don't correspond 

188 to any particular text. 

189 """ 

190 if not hasattr(node, 'first_token'): 

191 return (0, 0) 

192 

193 start = node.first_token.startpos 

194 if any(match_token(t, token.NEWLINE) for t in self.get_tokens(node)): 

195 # Multi-line nodes would be invalid unless we keep the indentation of the first node. 

196 start = self._text.rfind('\n', 0, start) + 1 

197 

198 return (start, node.last_token.endpos) 

199 

200 def get_text(self, node): 

201 """ 

202 After mark_tokens() has been called, returns the text corresponding to the given node. Returns 

203 '' for nodes (like `Load`) that don't correspond to any particular text. 

204 """ 

205 start, end = self.get_text_range(node) 

206 return self._text[start : end]
