52class MeasureBase:

58 def __init__(self, default_unit=None, **kwargs): 

59 value, self._default_unit = self.default_units(kwargs) 

60 setattr(self, self.STANDARD_UNIT, value) 

61 if default_unit and isinstance(default_unit, str): 

62 self._default_unit = default_unit

64 def _get_standard(self):

72 def __getattr__(self, name):

175 def default_units(self, kwargs): 

176 """ 

177 Return the unit value and the default units specified 

178 from the given keyword arguments dictionary. 

179 """ 

180 val = 0.0 

181 default_unit = self.STANDARD_UNIT 

182 for unit, value in kwargs.items(): 

183 if not isinstance(value, float): 

184 value = float(value) 

185 if unit in self.UNITS: 

186 val += self.UNITS[unit] * value 

187 default_unit = unit 

188 elif unit in self.ALIAS: 

189 u = self.ALIAS[unit] 

190 val += self.UNITS[u] * value 

191 default_unit = u 

192 else: 

193 lower = unit.lower() 

194 if lower in self.UNITS: 

195 val += self.UNITS[lower] * value 

196 default_unit = lower 

197 elif lower in self.LALIAS: 

198 u = self.LALIAS[lower] 

199 val += self.UNITS[u] * value 

200 default_unit = u 

201 else: 

202 raise AttributeError('Unknown unit type: %s' % unit) 

203 return val, default_unit

205 @classmethod 

206 def unit_attname(cls, unit_str): 

207 """ 

208 Retrieve the unit attribute name for the given unit string. 

209 For example, if the given unit string is 'metre', return 'm'. 

210 Raise an exception if an attribute cannot be found. 

211 """ 

212 lower = unit_str.lower() 

213 if unit_str in cls.UNITS: 

214 return unit_str 

215 elif lower in cls.UNITS: 

216 return lower 

217 elif lower in cls.LALIAS: 

218 return cls.LALIAS[lower] 

219 else: 

220 raise Exception('Could not find a unit keyword associated with "%s"' % unit_str)
