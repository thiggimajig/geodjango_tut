37def legend_scaler(legend_values, max_labels=10.0): 

38 """ 

39 Downsamples the number of legend values so that there isn't a collision 

40 of text on the legend colorbar (within reason). The colorbar seems to 

41 support ~10 entries as a maximum. 

42 

43 """ 

44 if len(legend_values) < max_labels: 

45 legend_ticks = legend_values 

46 else: 

47 spacer = int(math.ceil(len(legend_values)/max_labels)) 

48 legend_ticks = [] 

49 for i in legend_values[::spacer]: 

50 legend_ticks += [i] 

51 legend_ticks += ['']*(spacer-1) 

52 return legend_ticks

55def linear_gradient(hexList, nColors): 

56 """ 

57 Given a list of hexcode values, will return a list of length 

58 nColors where the colors are linearly interpolated between the 

59 (r, g, b) tuples that are given. 

60 

61 Examples 

62 -------- 

63 >>> linear_gradient([(0, 0, 0), (255, 0, 0), (255, 255, 0)], 100) 

64 

65 """ 

66 def _scale(start, finish, length, i): 

67 """ 

68 Return the value correct value of a number that is in between start 

69 and finish, for use in a loop of length *length*. 

70 

71 """ 

72 base = 16 

73 

74 fraction = float(i) / (length - 1) 

75 raynge = int(finish, base) - int(start, base) 

76 thex = hex(int(int(start, base) + fraction * raynge)).split('x')[-1] 

77 if len(thex) != 2: 

78 thex = '0' + thex 

79 return thex 

80 

81 allColors = [] 

82 # Separate (R, G, B) pairs. 

83 for start, end in zip(hexList[:-1], hexList[1:]): 

84 # Linearly intepolate between pair of hex ###### values and 

85 # add to list. 

86 nInterpolate = 765 

87 for index in range(nInterpolate): 

88 r = _scale(start[1:3], end[1:3], nInterpolate, index) 

89 g = _scale(start[3:5], end[3:5], nInterpolate, index) 

90 b = _scale(start[5:7], end[5:7], nInterpolate, index) 

91 allColors.append(''.join(['#', r, g, b])) 

92 

93 # Pick only nColors colors from the total list. 

94 result = [] 

95 for counter in range(nColors): 

96 fraction = float(counter) / (nColors - 1) 

97 index = int(fraction * (len(allColors) - 1)) 

98 result.append(allColors[index]) 

99 return result

102def color_brewer(color_code, n=6): 

103 """ 

104 Generate a colorbrewer color scheme of length 'len', type 'scheme. 

105 Live examples can be seen at http://colorbrewer2.org/ 

106 

107 """ 

108 maximum_n = 253 

109 minimum_n = 3 

110 

111 # Raise an error if the n requested is greater than the maximum. 

112 if n > maximum_n: 

113 raise ValueError('The maximum number of colors in a' 

114 ' ColorBrewer sequential color series is 253') 

115 if n < minimum_n: 

116 raise ValueError('The minimum number of colors in a' 

117 ' ColorBrewer sequential color series is 3') 

118 

119 if not isinstance(color_code, str): 

120 raise ValueError('color should be a string, not a {}.' 

121 .format(type(color_code))) 

122 if color_code[-2:] == '_r': 

123 base_code = color_code[:-2] 

124 core_color_code = base_code + '_' + str(n).zfill(2) 

125 color_reverse = True 

126 else: 

127 base_code = color_code 

128 core_color_code = base_code + '_' + str(n).zfill(2) 

129 color_reverse = False 

130 

131 with open(os.path.join(rootpath, '_schemes.json')) as f: 

132 schemes = json.loads(f.read()) 

133 

134 with open(os.path.join(rootpath, '_cnames.json')) as f: 

135 scheme_info = json.loads(f.read()) 

136 

137 with open(os.path.join(rootpath, 'scheme_base_codes.json')) as f: 

138 core_schemes = json.loads(f.read())['codes'] 

139 

140 if base_code not in core_schemes: 

141 raise ValueError(base_code + ' is not a valid ColorBrewer code') 

142 

143 try: 

144 schemes[core_color_code] 

145 explicit_scheme = True 

146 except KeyError: 

147 explicit_scheme = False 

148 

149 # Only if n is greater than the scheme length do we interpolate values. 

150 if not explicit_scheme: 

151 # Check to make sure that it is not a qualitative scheme. 

152 if scheme_info[base_code] == 'Qualitative': 

153 matching_quals = [] 

154 for key in schemes: 

155 if base_code + '_' in key: 

156 matching_quals.append(int(key.split('_')[1])) 

157 

158 raise ValueError('Expanded color support is not available' 

159 ' for Qualitative schemes; restrict the' 

160 ' number of colors for the ' + base_code + 

161 ' code to between ' + str(min(matching_quals)) + 

162 ' and ' + str(max(matching_quals)) 

163 ) 

164 else: 

165 if not color_reverse: 

166 color_scheme = linear_gradient(schemes.get(core_color_code), n) 

167 else: 

168 color_scheme = linear_gradient(schemes.get(core_color_code)[::-1], n) 

169 else: 

170 if not color_reverse: 

171 color_scheme = schemes.get(core_color_code, None) 

172 else: 

173 color_scheme = schemes.get(core_color_code, None)[::-1] 

174 return color_scheme

178 """ 

179 Given a Pandas Series, get a domain of values from zero to the 90% quantile 

180 rounded to the nearest order-of-magnitude integer. For example, 2100 is 

181 rounded to 2000, 2790 to 3000.

183 Parameters

185 series: Pandas series, default None

187 Returns

189 list

191 """

212def image_to_url(image, colormap=None, origin='upper'): 

213 """Infers the type of an image argument and transforms it into a URL. 

214 

215 Parameters 

216 ---------- 

217 image: string, file or array-like object 

218 * If string, it will be written directly in the output file. 

219 * If file, it's content will be converted as embedded in the 

220 output file. 

221 * If array-like, it will be converted to PNG base64 string and 

222 embedded in the output. 

223 origin : ['upper' | 'lower'], optional, default 'upper' 

224 Place the [0, 0] index of the array in the upper left or 

225 lower left corner of the axes. 

226 colormap : callable, used only for `mono` image. 

227 Function of the form [x -> (r,g,b)] or [x -> (r,g,b,a)] 

228 for transforming a mono image into RGB. 

229 It must output iterables of length 3 or 4, with values between 

230 0. and 1. Hint : you can use colormaps from `matplotlib.cm`. 

231 """ 

232 if hasattr(image, 'read'): 

233 # We got an image file. 

234 if hasattr(image, 'name'): 

235 # We try to get the image format from the file name. 

236 fileformat = image.name.lower().split('.')[-1] 

237 else: 

238 fileformat = 'png' 

239 url = 'data:image/{};base64,{}'.format( 

240 fileformat, base64.b64encode(image.read()).decode('utf-8')) 

241 elif (not (isinstance(image, str) or 

242 isinstance(image, bytes))) and hasattr(image, '__iter__'): 

243 # We got an array-like object. 

244 png = write_png(image, origin=origin, colormap=colormap) 

245 url = 'data:image/png;base64,' + base64.b64encode(png).decode('utf-8') 

246 else: 

247 # We got an URL. 

248 url = json.loads(json.dumps(image)) 

249 

250 return url.replace('\n', ' ')

253def write_png(data, origin='upper', colormap=None): 

254 """ 

255 Transform an array of data into a PNG string. 

256 This can be written to disk using binary I/O, or encoded using base64 

257 for an inline PNG like this: 

258 

259 >>> png_str = write_png(array) 

260 >>> 'data:image/png;base64,'+png_str.encode('base64') 

261 

262 Inspired from 

263 http://stackoverflow.com/questions/902761/saving-a-numpy-array-as-an-image 

264 

265 Parameters 

266 ---------- 

267 data: numpy array or equivalent list-like object. 

268 Must be NxM (mono), NxMx3 (RGB) or NxMx4 (RGBA) 

269 

270 origin : ['upper' | 'lower'], optional, default 'upper' 

271 Place the [0,0] index of the array in the upper left or lower left 

272 corner of the axes. 

273 

274 colormap : callable, used only for `mono` image. 

275 Function of the form [x -> (r,g,b)] or [x -> (r,g,b,a)] 

276 for transforming a mono image into RGB. 

277 It must output iterables of length 3 or 4, with values between 

278 0. and 1. Hint: you can use colormaps from `matplotlib.cm`. 

279 

280 Returns 

281 ------- 

282 PNG formatted byte string 

283 """ 

284 if np is None: 

285 raise ImportError('The NumPy package is required' 

286 ' for this functionality') 

287 

288 if colormap is None: 

289 def colormap(x): 

290 return (x, x, x, 1) 

291 

292 array = np.atleast_3d(data) 

293 height, width, nblayers = array.shape 

294 

295 if nblayers not in [1, 3, 4]: 

296 raise ValueError('Data must be NxM (mono), ' 

297 'NxMx3 (RGB), or NxMx4 (RGBA)') 

298 assert array.shape == (height, width, nblayers) 

299 

300 if nblayers == 1: 

301 array = np.array(list(map(colormap, array.ravel()))) 

302 nblayers = array.shape[1] 

303 if nblayers not in [3, 4]: 

304 raise ValueError('colormap must provide colors of' 

305 'length 3 (RGB) or 4 (RGBA)') 

306 array = array.reshape((height, width, nblayers)) 

307 assert array.shape == (height, width, nblayers) 

308 

309 if nblayers == 3: 

310 array = np.concatenate((array, np.ones((height, width, 1))), axis=2) 

311 nblayers = 4 

312 assert array.shape == (height, width, nblayers) 

313 assert nblayers == 4 

314 

315 # Normalize to uint8 if it isn't already. 

316 if array.dtype != 'uint8': 

317 array = array * 255./array.max(axis=(0, 1)).reshape((1, 1, 4)) 

318 array = array.astype('uint8') 

319 

320 # Eventually flip the image. 

321 if origin == 'lower': 

322 array = array[::-1, :, :] 

323 

324 # Transform the array to bytes. 

325 raw_data = b''.join([b'\x00' + array[i, :, :].tobytes() 

326 for i in range(height)])

328 def png_pack(png_tag, data): 

329 chunk_head = png_tag + data 

330 return (struct.pack('!I', len(data)) + 

331 chunk_head + 

332 struct.pack('!I', 0xFFFFFFFF & zlib.crc32(chunk_head))) 

333 

334 return b''.join([ 

335 b'\x89PNG\r\n\x1a\n', 

336 png_pack(b'IHDR', struct.pack('!2I5B', width, height, 8, 6, 0, 0, 0)), 

337 png_pack(b'IDAT', zlib.compress(raw_data, 9)), 

338 png_pack(b'IEND', b'')])
