13@contextmanager 

14def assert_produces_warning( 

15 expected_warning: type[Warning] | bool | None = Warning, 

16 filter_level="always", 

17 check_stacklevel: bool = True, 

18 raise_on_extra_warnings: bool = True, 

19 match: str | None = None, 

20): 

21 """ 

22 Context manager for running code expected to either raise a specific 

23 warning, or not raise any warnings. Verifies that the code raises the 

24 expected warning, and that it does not raise any other unexpected 

25 warnings. It is basically a wrapper around ``warnings.catch_warnings``. 

26 

27 Parameters 

28 ---------- 

29 expected_warning : {Warning, False, None}, default Warning 

30 The type of Exception raised. ``exception.Warning`` is the base 

31 class for all warnings. To check that no warning is returned, 

32 specify ``False`` or ``None``. 

33 filter_level : str or None, default "always" 

34 Specifies whether warnings are ignored, displayed, or turned 

35 into errors. 

36 Valid values are: 

37 

38 * "error" - turns matching warnings into exceptions 

39 * "ignore" - discard the warning 

40 * "always" - always emit a warning 

41 * "default" - print the warning the first time it is generated 

42 from each location 

43 * "module" - print the warning the first time it is generated 

44 from each module 

45 * "once" - print the warning the first time it is generated 

46 

47 check_stacklevel : bool, default True 

48 If True, displays the line that called the function containing 

49 the warning to show were the function is called. Otherwise, the 

50 line that implements the function is displayed. 

51 raise_on_extra_warnings : bool, default True 

52 Whether extra warnings not of the type `expected_warning` should 

53 cause the test to fail. 

54 match : str, optional 

55 Match warning message. 

56 

57 Examples 

58 -------- 

59 >>> import warnings 

60 >>> with assert_produces_warning(): 

61 ... warnings.warn(UserWarning()) 

62 ... 

63 >>> with assert_produces_warning(False): 

64 ... warnings.warn(RuntimeWarning()) 

65 ... 

66 Traceback (most recent call last): 

67 ... 

68 AssertionError: Caused unexpected warning(s): ['RuntimeWarning']. 

69 >>> with assert_produces_warning(UserWarning): 

70 ... warnings.warn(RuntimeWarning()) 

71 Traceback (most recent call last): 

72 ... 

73 AssertionError: Did not see expected warning of class 'UserWarning'. 

74 

75 ..warn:: This is *not* thread-safe. 

76 """

99def _assert_caught_expected_warning( 

100 *, 

101 caught_warnings: Sequence[warnings.WarningMessage], 

102 expected_warning: type[Warning], 

103 match: str | None, 

104 check_stacklevel: bool, 

105) -> None: 

106 """Assert that there was the expected warning among the caught warnings.""" 

107 saw_warning = False 

108 matched_message = False 

109 

110 for actual_warning in caught_warnings: 

111 if issubclass(actual_warning.category, expected_warning): 

112 saw_warning = True 

113 

114 if check_stacklevel and issubclass( 

115 actual_warning.category, (FutureWarning, DeprecationWarning) 

116 ): 

117 _assert_raised_with_correct_stacklevel(actual_warning) 

118 

119 if match is not None and re.search(match, str(actual_warning.message)): 

120 matched_message = True 

121 

122 if not saw_warning: 

123 raise AssertionError( 

124 f"Did not see expected warning of class " 

125 f"{repr(expected_warning.__name__)}" 

126 ) 

127 

128 if match and not matched_message: 

129 raise AssertionError( 

130 f"Did not see warning {repr(expected_warning.__name__)} " 

131 f"matching {match}" 

132 )

166def _is_unexpected_warning( 

167 actual_warning: warnings.WarningMessage, 

168 expected_warning: type[Warning] | bool | None, 

169) -> bool: 

170 """Check if the actual warning issued is unexpected.""" 

171 if actual_warning and not expected_warning: 

172 return True 

173 expected_warning = cast(Type[Warning], expected_warning) 

174 return bool(not issubclass(actual_warning.category, expected_warning))
