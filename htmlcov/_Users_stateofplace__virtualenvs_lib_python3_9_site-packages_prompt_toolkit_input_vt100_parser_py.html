16# Regex matching any CPR response 

17# (Note that we use '\Z' instead of '$', because '$' could include a trailing 

18# newline.) 

19_cpr_response_re = re.compile("^" + re.escape("\x1b[") + r"\d+;\d+R\Z")

21# Mouse events: 

22# Typical: "Esc[MaB*" Urxvt: "Esc[96;14;13M" and for Xterm SGR: "Esc[<64;85;12M" 

23_mouse_event_re = re.compile("^" + re.escape("\x1b[") + r"(<?[\d;]+[mM]|M...)\Z")

101 def _get_match(self, prefix: str) -> Union[None, Keys, Tuple[Keys, ...]]: 

102 """ 

103 Return the key (or keys) that maps to this prefix. 

104 """ 

105 # (hard coded) If we match a CPR response, return Keys.CPRResponse. 

106 # (This one doesn't fit in the ANSI_SEQUENCES, because it contains 

107 # integer variables.) 

108 if _cpr_response_re.match(prefix): 

109 return Keys.CPRResponse 

110 

111 elif _mouse_event_re.match(prefix): 

112 return Keys.Vt100MouseEvent 

113 

114 # Otherwise, use the mappings. 

115 try: 

116 return ANSI_SEQUENCES[prefix] 

117 except KeyError: 

118 return None

120 def _input_parser_generator(self) -> Generator[None, Union[str, _Flush], None]: 

121 """ 

122 Coroutine (state machine) for the input parser. 

123 """ 

124 prefix = "" 

125 retry = False 

126 flush = False 

127 

128 while True: 

129 flush = False 

130 

131 if retry: 

132 retry = False 

133 else: 

134 # Get next character. 

135 c = yield 

136 

137 if isinstance(c, _Flush): 

138 flush = True 

139 else: 

140 prefix += c 

141 

142 # If we have some data, check for matches. 

143 if prefix: 

144 is_prefix_of_longer_match = _IS_PREFIX_OF_LONGER_MATCH_CACHE[prefix] 

145 match = self._get_match(prefix) 

146 

147 # Exact matches found, call handlers.. 

148 if (flush or not is_prefix_of_longer_match) and match: 

149 self._call_handler(match, prefix) 

150 prefix = "" 

151 

152 # No exact match found. 

153 elif (flush or not is_prefix_of_longer_match) and not match: 

154 found = False 

155 retry = True 

156 

157 # Loop over the input, try the longest match first and 

158 # shift. 

159 for i in range(len(prefix), 0, -1): 

160 match = self._get_match(prefix[:i]) 

161 if match: 

162 self._call_handler(match, prefix[:i]) 

163 prefix = prefix[i:] 

164 found = True 

165 

166 if not found: 

167 self._call_handler(prefix[0], prefix[0]) 

168 prefix = prefix[1:]

170 def _call_handler( 

171 self, key: Union[str, Keys, Tuple[Keys, ...]], insert_text: str

173 """ 

174 Callback to handler. 

175 """ 

176 if isinstance(key, tuple): 

177 # Received ANSI sequence that corresponds with multiple keys 

178 # (probably alt+something). Handle keys individually, but only pass 

179 # data payload to first KeyPress (so that we won't insert it 

180 # multiple times).

185 self._in_bracketed_paste = True 

186 self._paste_buffer = ""

190 def feed(self, data: str) -> None: 

191 """ 

192 Feed the input stream. 

193 

194 :param data: Input string (unicode). 

195 """ 

196 # Handle bracketed paste. (We bypass the parser that matches all other 

197 # key presses and keep reading input until we see the end mark.) 

198 # This is much faster then parsing character by character. 

199 if self._in_bracketed_paste: 

200 self._paste_buffer += data 

201 end_mark = "\x1b[201~" 

202 

203 if end_mark in self._paste_buffer: 

204 end_index = self._paste_buffer.index(end_mark) 

205 

206 # Feed content to key bindings. 

207 paste_content = self._paste_buffer[:end_index] 

208 self.feed_key_callback(KeyPress(Keys.BracketedPaste, paste_content)) 

209 

210 # Quit bracketed paste mode and handle remaining input. 

211 self._in_bracketed_paste = False 

212 remaining = self._paste_buffer[end_index + len(end_mark) :] 

213 self._paste_buffer = "" 

214 

215 self.feed(remaining) 

216 

217 # Handle normal input character by character. 

218 else: 

219 for i, c in enumerate(data): 

220 if self._in_bracketed_paste: 

221 # Quit loop and process from this position when the parser 

222 # entered bracketed paste. 

223 self.feed(data[i:]) 

224 break 

225 else: 

226 self._input_parser.send(c)

228 def flush(self) -> None: 

229 """ 

230 Flush the buffer of the input stream. 

231 

232 This will allow us to handle the escape key (or maybe meta) sooner. 

233 The input received by the escape key is actually the same as the first 

234 characters of e.g. Arrow-Up, so without knowing what follows the escape 

235 sequence, we don't know whether escape has been pressed, or whether 

236 it's something else. This flush function should be called after a 

237 timeout, and processes everything that's still in the buffer as-is, so 

238 without assuming any characters will follow. 

239 """ 

240 self._input_parser.send(_Flush())
