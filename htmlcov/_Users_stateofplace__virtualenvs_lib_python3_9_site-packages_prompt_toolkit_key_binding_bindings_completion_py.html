32def generate_completions(event: E) -> None: 

33 r""" 

34 Tab-completion: where the first tab completes the common suffix and the 

35 second tab lists all the completions. 

36 """ 

37 b = event.current_buffer 

38 

39 # When already navigating through completions, select the next one. 

40 if b.complete_state: 

41 b.complete_next() 

42 else: 

43 b.start_completion(insert_common_part=True)

46def display_completions_like_readline(event: E) -> None: 

47 """ 

48 Key binding handler for readline-style tab completion. 

49 This is meant to be as similar as possible to the way how readline displays 

50 completions. 

51 

52 Generate the completions immediately (blocking) and display them above the 

53 prompt in columns. 

54 

55 Usage:: 

56 

57 # Call this handler when 'Tab' has been pressed. 

58 key_bindings.add(Keys.ControlI)(display_completions_like_readline) 

59 """ 

60 # Request completions. 

61 b = event.current_buffer 

62 if b.completer is None: 

63 return 

64 complete_event = CompleteEvent(completion_requested=True) 

65 completions = list(b.completer.get_completions(b.document, complete_event)) 

66 

67 # Calculate the common suffix. 

68 common_suffix = get_common_complete_suffix(b.document, completions) 

69 

70 # One completion: insert it. 

71 if len(completions) == 1: 

72 b.delete_before_cursor(-completions[0].start_position) 

73 b.insert_text(completions[0].text) 

74 # Multiple completions with common part. 

75 elif common_suffix: 

76 b.insert_text(common_suffix) 

77 # Otherwise: display all completions. 

78 elif completions: 

79 _display_completions_like_readline(event.app, completions)

82def _display_completions_like_readline( 

83 app: "Application[object]", completions: List[Completion] 

84) -> "asyncio.Task[None]": 

85 """ 

86 Display the list of completions in columns above the prompt. 

87 This will ask for a confirmation if there are too many completions to fit 

88 on a single page and provide a paginator to walk through them. 

89 """ 

90 from prompt_toolkit.formatted_text import to_formatted_text 

91 from prompt_toolkit.shortcuts.prompt import create_confirm_session 

92 

93 # Get terminal dimensions. 

94 term_size = app.output.get_size() 

95 term_width = term_size.columns 

96 term_height = term_size.rows 

97 

98 # Calculate amount of required columns/rows for displaying the 

99 # completions. (Keep in mind that completions are displayed 

100 # alphabetically column-wise.) 

101 max_compl_width = min( 

102 term_width, max(get_cwidth(c.display_text) for c in completions) + 1 

103 ) 

104 column_count = max(1, term_width // max_compl_width) 

105 completions_per_page = column_count * (term_height - 1) 

106 page_count = int(math.ceil(len(completions) / float(completions_per_page))) 

107 # Note: math.ceil can return float on Python2. 

108 

109 def display(page: int) -> None: 

110 # Display completions. 

111 page_completions = completions[ 

112 page * completions_per_page : (page + 1) * completions_per_page 

113 ] 

114 

115 page_row_count = int(math.ceil(len(page_completions) / float(column_count))) 

116 page_columns = [ 

117 page_completions[i * page_row_count : (i + 1) * page_row_count] 

118 for i in range(column_count) 

119 ] 

120 

121 result: StyleAndTextTuples = [] 

122 

123 for r in range(page_row_count): 

124 for c in range(column_count): 

125 try: 

126 completion = page_columns[c][r] 

127 style = "class:readline-like-completions.completion " + ( 

128 completion.style or "" 

129 ) 

130 

131 result.extend(to_formatted_text(completion.display, style=style)) 

132 

133 # Add padding. 

134 padding = max_compl_width - get_cwidth(completion.display_text) 

135 result.append((completion.style, " " * padding)) 

136 except IndexError: 

137 pass 

138 result.append(("", "\n")) 

139 

140 app.print_text(to_formatted_text(result, "class:readline-like-completions")) 

141 

142 # User interaction through an application generator function. 

143 async def run_compl() -> None: 

144 "Coroutine." 

145 async with in_terminal(render_cli_done=True): 

146 if len(completions) > completions_per_page: 

147 # Ask confirmation if it doesn't fit on the screen. 

148 confirm = await create_confirm_session( 

149 "Display all {} possibilities?".format(len(completions)), 

150 ).prompt_async() 

151 

152 if confirm: 

153 # Display pages. 

154 for page in range(page_count): 

155 display(page) 

156 

157 if page != page_count - 1: 

158 # Display --MORE-- and go to the next page. 

159 show_more = await _create_more_session( 

160 "--MORE--" 

161 ).prompt_async() 

162 

163 if not show_more: 

164 return 

165 else: 

166 app.output.flush() 

167 else: 

168 # Display all completions. 

169 display(0) 

170 

171 return app.create_background_task(run_compl())
