183def load_mouse_bindings() -> KeyBindings: 

184 """ 

185 Key bindings, required for mouse support. 

186 (Mouse events enter through the key binding system.) 

187 """ 

188 key_bindings = KeyBindings() 

189 

190 @key_bindings.add(Keys.Vt100MouseEvent) 

191 def _(event: E) -> "NotImplementedOrNone": 

192 """ 

193 Handling of incoming mouse event. 

194 """ 

195 # TypicaL: "eSC[MaB*" 

196 # Urxvt: "Esc[96;14;13M" 

197 # Xterm SGR: "Esc[<64;85;12M" 

198 

199 # Parse incoming packet. 

200 if event.data[2] == "M": 

201 # Typical. 

202 mouse_event, x, y = map(ord, event.data[3:]) 

203 

204 # TODO: Is it possible to add modifiers here? 

205 mouse_button, mouse_event_type, mouse_modifier = typical_mouse_events[ 

206 mouse_event 

207 ] 

208 

209 # Handle situations where `PosixStdinReader` used surrogateescapes. 

210 if x >= 0xDC00: 

211 x -= 0xDC00 

212 if y >= 0xDC00: 

213 y -= 0xDC00 

214 

215 x -= 32 

216 y -= 32 

217 else: 

218 # Urxvt and Xterm SGR. 

219 # When the '<' is not present, we are not using the Xterm SGR mode, 

220 # but Urxvt instead. 

221 data = event.data[2:] 

222 if data[:1] == "<": 

223 sgr = True 

224 data = data[1:] 

225 else: 

226 sgr = False 

227 

228 # Extract coordinates. 

229 mouse_event, x, y = map(int, data[:-1].split(";")) 

230 m = data[-1] 

231 

232 # Parse event type. 

233 if sgr: 

234 try: 

235 ( 

236 mouse_button, 

237 mouse_event_type, 

238 mouse_modifiers, 

239 ) = xterm_sgr_mouse_events[mouse_event, m] 

240 except KeyError: 

241 return NotImplemented 

242 

243 else: 

244 # Some other terminals, like urxvt, Hyper terminal, ... 

245 ( 

246 mouse_button, 

247 mouse_event_type, 

248 mouse_modifiers, 

249 ) = urxvt_mouse_events.get( 

250 mouse_event, (UNKNOWN_BUTTON, MOUSE_MOVE, UNKNOWN_MODIFIER) 

251 ) 

252 

253 x -= 1 

254 y -= 1 

255 

256 # Only handle mouse events when we know the window height. 

257 if event.app.renderer.height_is_known and mouse_event_type is not None: 

258 # Take region above the layout into account. The reported 

259 # coordinates are absolute to the visible part of the terminal. 

260 from prompt_toolkit.renderer import HeightIsUnknownError 

261 

262 try: 

263 y -= event.app.renderer.rows_above_layout 

264 except HeightIsUnknownError: 

265 return NotImplemented 

266 

267 # Call the mouse handler from the renderer. 

268 

269 # Note: This can return `NotImplemented` if no mouse handler was 

270 # found for this position, or if no repainting needs to 

271 # happen. this way, we avoid excessive repaints during mouse 

272 # movements. 

273 handler = event.app.renderer.mouse_handlers.mouse_handlers[y][x] 

274 return handler( 

275 MouseEvent( 

276 position=Point(x=x, y=y), 

277 event_type=mouse_event_type, 

278 button=mouse_button, 

279 modifiers=mouse_modifiers, 

280 ) 

281 ) 

282 

283 return NotImplemented

287 """ 

288 Scroll up event without cursor position.

301 @key_bindings.add(Keys.WindowsMouseEvent) 

302 def _mouse(event: E) -> "NotImplementedOrNone": 

303 """ 

304 Handling of mouse events for Windows. 

305 """ 

306 assert is_windows() # This key binding should only exist for Windows. 

307 

308 # Parse data. 

309 pieces = event.data.split(";") 

310 

311 button = MouseButton(pieces[0]) 

312 event_type = MouseEventType(pieces[1]) 

313 x = int(pieces[2]) 

314 y = int(pieces[3]) 

315 

316 # Make coordinates absolute to the visible part of the terminal. 

317 output = event.app.renderer.output 

318 

319 from prompt_toolkit.output.win32 import Win32Output 

320 from prompt_toolkit.output.windows10 import Windows10_Output 

321 

322 if isinstance(output, (Win32Output, Windows10_Output)): 

323 screen_buffer_info = output.get_win32_screen_buffer_info() 

324 rows_above_cursor = ( 

325 screen_buffer_info.dwCursorPosition.Y - event.app.renderer._cursor_pos.y 

326 ) 

327 y -= rows_above_cursor 

328 

329 # Call the mouse event handler. 

330 # (Can return `NotImplemented`.) 

331 handler = event.app.renderer.mouse_handlers.mouse_handlers[y][x] 

332 

333 return handler( 

334 MouseEvent( 

335 position=Point(x=x, y=y), 

336 event_type=event_type, 

337 button=button, 

338 modifiers=UNKNOWN_MODIFIER, 

339 ) 

340 ) 

341 

342 # No mouse handler found. Return `NotImplemented` so that we don't 

343 # invalidate the UI. 

344 return NotImplemented 

345 

346 return key_bindings
