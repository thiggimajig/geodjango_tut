159def _is_aligned(frame, other): 

160 """ 

161 Helper to check if a DataFrame is aligned with another DataFrame or Series. 

162 """ 

163 from pandas import DataFrame 

164 

165 if isinstance(other, DataFrame): 

166 return frame._indexed_same(other) 

167 else: 

168 # Series -> match index 

169 return frame.columns.equals(other.index)

172def _maybe_fallback(ufunc: np.ufunc, method: str, *inputs: Any, **kwargs: Any): 

173 """ 

174 In the future DataFrame, inputs to ufuncs will be aligned before applying 

175 the ufunc, but for now we ignore the index but raise a warning if behaviour 

176 would change in the future. 

177 This helper detects the case where a warning is needed and then fallbacks 

178 to applying the ufunc on arrays to avoid alignment. 

179 

180 See https://github.com/pandas-dev/pandas/pull/39239 

181 """ 

182 from pandas import DataFrame 

183 from pandas.core.generic import NDFrame 

184 

185 n_alignable = sum(isinstance(x, NDFrame) for x in inputs) 

186 n_frames = sum(isinstance(x, DataFrame) for x in inputs) 

187 

188 if n_alignable >= 2 and n_frames >= 1: 

189 # if there are 2 alignable inputs (Series or DataFrame), of which at least 1 

190 # is a DataFrame -> we would have had no alignment before -> warn that this 

191 # will align in the future 

192 

193 # the first frame is what determines the output index/columns in pandas < 1.2 

194 first_frame = next(x for x in inputs if isinstance(x, DataFrame)) 

195 

196 # check if the objects are aligned or not 

197 non_aligned = sum( 

198 not _is_aligned(first_frame, x) for x in inputs if isinstance(x, NDFrame) 

199 ) 

200 

201 # if at least one is not aligned -> warn and fallback to array behaviour 

202 if non_aligned: 

203 warnings.warn( 

204 "Calling a ufunc on non-aligned DataFrames (or DataFrame/Series " 

205 "combination). Currently, the indices are ignored and the result " 

206 "takes the index/columns of the first DataFrame. In the future , " 

207 "the DataFrames/Series will be aligned before applying the ufunc.\n" 

208 "Convert one of the arguments to a NumPy array " 

209 "(eg 'ufunc(df1, np.asarray(df2)') to keep the current behaviour, " 

210 "or align manually (eg 'df1, df2 = df1.align(df2)') before passing to " 

211 "the ufunc to obtain the future behaviour and silence this warning.", 

212 FutureWarning, 

213 stacklevel=4, 

214 ) 

215 

216 # keep the first dataframe of the inputs, other DataFrame/Series is 

217 # converted to array for fallback behaviour 

218 new_inputs = [] 

219 for x in inputs: 

220 if x is first_frame: 

221 new_inputs.append(x) 

222 elif isinstance(x, NDFrame): 

223 new_inputs.append(np.asarray(x)) 

224 else: 

225 new_inputs.append(x) 

226 

227 # call the ufunc on those transformed inputs 

228 return getattr(ufunc, method)(*new_inputs, **kwargs) 

229 

230 # signal that we didn't fallback / execute the ufunc yet 

231 return NotImplemented

234def array_ufunc(self, ufunc: np.ufunc, method: str, *inputs: Any, **kwargs: Any): 

235 """ 

236 Compatibility with numpy ufuncs. 

237 

238 See also 

239 -------- 

240 numpy.org/doc/stable/reference/arrays.classes.html#numpy.class.__array_ufunc__ 

241 """ 

242 from pandas.core.generic import NDFrame 

243 from pandas.core.internals import BlockManager 

244 

245 cls = type(self) 

246 

247 # for backwards compatibility check and potentially fallback for non-aligned frames 

248 result = _maybe_fallback(ufunc, method, *inputs, **kwargs) 

249 if result is not NotImplemented: 

250 return result 

251 

252 # for binary ops, use our custom dunder methods 

253 result = maybe_dispatch_ufunc_to_dunder_op(self, ufunc, method, *inputs, **kwargs) 

254 if result is not NotImplemented: 

255 return result 

256 

257 # Determine if we should defer. 

258 

259 # error: "Type[ndarray]" has no attribute "__array_ufunc__" 

260 no_defer = ( 

261 np.ndarray.__array_ufunc__, # type: ignore[attr-defined] 

262 cls.__array_ufunc__, 

263 ) 

264 

265 for item in inputs: 

266 higher_priority = ( 

267 hasattr(item, "__array_priority__") 

268 and item.__array_priority__ > self.__array_priority__ 

269 ) 

270 has_array_ufunc = ( 

271 hasattr(item, "__array_ufunc__") 

272 and type(item).__array_ufunc__ not in no_defer 

273 and not isinstance(item, self._HANDLED_TYPES) 

274 ) 

275 if higher_priority or has_array_ufunc: 

276 return NotImplemented 

277 

278 # align all the inputs. 

279 types = tuple(type(x) for x in inputs) 

280 alignable = [x for x, t in zip(inputs, types) if issubclass(t, NDFrame)] 

281 

282 if len(alignable) > 1: 

283 # This triggers alignment. 

284 # At the moment, there aren't any ufuncs with more than two inputs 

285 # so this ends up just being x1.index | x2.index, but we write 

286 # it to handle *args. 

287 

288 if len(set(types)) > 1: 

289 # We currently don't handle ufunc(DataFrame, Series) 

290 # well. Previously this raised an internal ValueError. We might 

291 # support it someday, so raise a NotImplementedError. 

292 raise NotImplementedError( 

293 "Cannot apply ufunc {} to mixed DataFrame and Series " 

294 "inputs.".format(ufunc) 

295 ) 

296 axes = self.axes 

297 for obj in alignable[1:]: 

298 # this relies on the fact that we aren't handling mixed 

299 # series / frame ufuncs. 

300 for i, (ax1, ax2) in enumerate(zip(axes, obj.axes)): 

301 axes[i] = ax1.union(ax2) 

302 

303 reconstruct_axes = dict(zip(self._AXIS_ORDERS, axes)) 

304 inputs = tuple( 

305 x.reindex(**reconstruct_axes) if issubclass(t, NDFrame) else x 

306 for x, t in zip(inputs, types) 

307 ) 

308 else: 

309 reconstruct_axes = dict(zip(self._AXIS_ORDERS, self.axes)) 

310 

311 if self.ndim == 1: 

312 names = [getattr(x, "name") for x in inputs if hasattr(x, "name")] 

313 name = names[0] if len(set(names)) == 1 else None 

314 reconstruct_kwargs = {"name": name} 

315 else: 

316 reconstruct_kwargs = {} 

317 

318 def reconstruct(result): 

319 if lib.is_scalar(result): 

320 return result 

321 if result.ndim != self.ndim: 

322 if method == "outer": 

323 if self.ndim == 2: 

324 # we already deprecated for Series 

325 msg = ( 

326 "outer method for ufunc {} is not implemented on " 

327 "pandas objects. Returning an ndarray, but in the " 

328 "future this will raise a 'NotImplementedError'. " 

329 "Consider explicitly converting the DataFrame " 

330 "to an array with '.to_numpy()' first." 

331 ) 

332 warnings.warn(msg.format(ufunc), FutureWarning, stacklevel=4) 

333 return result 

334 raise NotImplementedError 

335 return result 

336 if isinstance(result, BlockManager): 

337 # we went through BlockManager.apply 

338 result = self._constructor(result, **reconstruct_kwargs, copy=False) 

339 else: 

340 # we converted an array, lost our axes 

341 result = self._constructor( 

342 result, **reconstruct_axes, **reconstruct_kwargs, copy=False 

343 ) 

344 # TODO: When we support multiple values in __finalize__, this 

345 # should pass alignable to `__finalize__` instead of self. 

346 # Then `np.add(a, b)` would consider attrs from both a and b 

347 # when a and b are NDFrames. 

348 if len(alignable) == 1: 

349 result = result.__finalize__(self) 

350 return result 

351 

352 if self.ndim > 1 and (len(inputs) > 1 or ufunc.nout > 1): 

353 # Just give up on preserving types in the complex case. 

354 # In theory we could preserve them for them. 

355 # * nout>1 is doable if BlockManager.apply took nout and 

356 # returned a Tuple[BlockManager]. 

357 # * len(inputs) > 1 is doable when we know that we have 

358 # aligned blocks / dtypes. 

359 inputs = tuple(np.asarray(x) for x in inputs) 

360 result = getattr(ufunc, method)(*inputs, **kwargs) 

361 elif self.ndim == 1: 

362 # ufunc(series, ...) 

363 inputs = tuple(extract_array(x, extract_numpy=True) for x in inputs) 

364 result = getattr(ufunc, method)(*inputs, **kwargs) 

365 else: 

366 # ufunc(dataframe) 

367 if method == "__call__" and not kwargs: 

368 # for np.<ufunc>(..) calls 

369 # kwargs cannot necessarily be handled block-by-block, so only 

370 # take this path if there are no kwargs 

371 mgr = inputs[0]._mgr 

372 result = mgr.apply(getattr(ufunc, method)) 

373 else: 

374 # otherwise specific ufunc methods (eg np.<ufunc>.accumulate(..)) 

375 # Those can have an axis keyword and thus can't be called block-by-block 

376 result = getattr(ufunc, method)(np.asarray(inputs[0]), **kwargs) 

377 

378 if ufunc.nout > 1: 

379 result = tuple(reconstruct(x) for x in result) 

380 else: 

381 result = reconstruct(result) 

382 return result
