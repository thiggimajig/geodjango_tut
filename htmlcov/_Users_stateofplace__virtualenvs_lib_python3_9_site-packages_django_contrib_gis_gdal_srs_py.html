45 """ 

46 A wrapper for the OGRSpatialReference object. According to the GDAL Web site, 

47 the SpatialReference object "provide[s] services to represent coordinate 

48 systems (projections and datums) and to transform between them."

52 def __init__(self, srs_input='', srs_type='user', axis_order=None): 

53 """ 

54 Create a GDAL OSR Spatial Reference object from the given input. 

55 The input may be string of OGC Well Known Text (WKT), an integer 

56 EPSG code, a PROJ string, and/or a projection "well known" shorthand 

57 string (one of 'WGS84', 'WGS72', 'NAD27', 'NAD83'). 

58 """ 

59 if not isinstance(axis_order, (type(None), AxisOrder)): 

60 raise ValueError( 

61 'SpatialReference.axis_order must be an AxisOrder instance.' 

62 ) 

63 self.axis_order = axis_order or AxisOrder.TRADITIONAL 

64 if srs_type == 'wkt': 

65 self.ptr = capi.new_srs(c_char_p(b'')) 

66 self.import_wkt(srs_input) 

67 if self.axis_order == AxisOrder.TRADITIONAL and GDAL_VERSION >= (3, 0): 

68 capi.set_axis_strategy(self.ptr, self.axis_order) 

69 elif self.axis_order != AxisOrder.TRADITIONAL and GDAL_VERSION < (3, 0): 

70 raise ValueError('%s is not supported in GDAL < 3.0.' % self.axis_order) 

71 return 

72 elif isinstance(srs_input, str): 

73 try: 

74 # If SRID is a string, e.g., '4326', then make acceptable 

75 # as user input. 

76 srid = int(srs_input) 

77 srs_input = 'EPSG:%d' % srid 

78 except ValueError: 

79 pass 

80 elif isinstance(srs_input, int): 

81 # EPSG integer code was input. 

82 srs_type = 'epsg' 

83 elif isinstance(srs_input, self.ptr_type): 

84 srs = srs_input 

85 srs_type = 'ogr' 

86 else: 

87 raise TypeError('Invalid SRS type "%s"' % srs_type) 

88 

89 if srs_type == 'ogr': 

90 # Input is already an SRS pointer. 

91 srs = srs_input 

92 else: 

93 # Creating a new SRS pointer, using the string buffer. 

94 buf = c_char_p(b'') 

95 srs = capi.new_srs(buf) 

96 

97 # If the pointer is NULL, throw an exception. 

98 if not srs: 

99 raise SRSException('Could not create spatial reference from: %s' % srs_input) 

100 else: 

101 self.ptr = srs 

102 

103 if self.axis_order == AxisOrder.TRADITIONAL and GDAL_VERSION >= (3, 0): 

104 capi.set_axis_strategy(self.ptr, self.axis_order) 

105 elif self.axis_order != AxisOrder.TRADITIONAL and GDAL_VERSION < (3, 0): 

106 raise ValueError('%s is not supported in GDAL < 3.0.' % self.axis_order) 

107 # Importing from either the user input string or an integer SRID. 

108 if srs_type == 'user': 

109 self.import_user_input(srs_input) 

110 elif srs_type == 'epsg': 

111 self.import_epsg(srs_input)

113 def __getitem__(self, target): 

114 """ 

115 Return the value of the given string attribute node, None if the node 

116 doesn't exist. Can also take a tuple as a parameter, (target, child), 

117 where child is the index of the attribute in the WKT. For example: 

118 

119 >>> wkt = 'GEOGCS["WGS 84", DATUM["WGS_1984, ... AUTHORITY["EPSG","4326"]]' 

120 >>> srs = SpatialReference(wkt) # could also use 'WGS84', or 4326 

121 >>> print(srs['GEOGCS']) 

122 WGS 84 

123 >>> print(srs['DATUM']) 

124 WGS_1984 

125 >>> print(srs['AUTHORITY']) 

126 EPSG 

127 >>> print(srs['AUTHORITY', 1]) # The authority value 

128 4326 

129 >>> print(srs['TOWGS84', 4]) # the fourth value in this wkt 

130 0 

131 >>> print(srs['UNIT|AUTHORITY']) # For the units authority, have to use the pipe symbole. 

132 EPSG 

133 >>> print(srs['UNIT|AUTHORITY', 1]) # The authority value for the units 

134 9122 

135 """ 

136 if isinstance(target, tuple): 

137 return self.attr_value(*target) 

138 else: 

139 return self.attr_value(target)
